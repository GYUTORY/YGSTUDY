    Nginx Rate Limiting¶

    к°ңмҡ”¶

    Rate LimitingмқҖ нҒҙлқјмқҙм–ёнҠёк°Җ мқјм • мӢңк°„ лҸҷм•Ҳ ліҙлӮј мҲҳ мһҲлҠ” мҡ”мІӯ мҲҳлҘј м ңн•ңн•ҳлҠ” кё°лІ•мқҙлӢӨ. л¬ҙм°Ёлі„ лҢҖмһ… кіөкІ©, DDoS, API лӮЁмҡ©мқ„ л°©м–ҙн•ҳлҠ” мІ« лІҲм§ё л°©м–ҙм„ мқҙлӢӨ.

    Rate Limitingмқҙ н•„мҡ”н•ң мғҒнҷ©:
  - лЎңк·ёмқё м—”л“ңнҸ¬мқёнҠём—җ brute force кіөкІ©
  - нҠ№м • IPк°Җ мҙҲлӢ№ мҲҳл°ұ к°ң мҡ”мІӯмқ„ ліҙлӮҙ м„ңлІ„ л¶Җн•ҳ мң л°ң
  - мҷёл¶Җ кіөк°ң APIмқҳ кіјлҸ„н•ң мӮ¬мҡ© м ңн•ң
  - нҒ¬лЎӨлҹ¬/лҙҮ нҠёлһҳн”Ҫ м ңм–ҙ

    Leaky Bucket м•Ңкі лҰ¬мҰҳ вҖ” limit_reqмқҳ лӮҙл¶Җ лҸҷмһ‘¶

    Nginxмқҳ limit_reqлҠ” Leaky Bucket(лҲ„мҲҳ лІ„нӮ·) м•Ңкі лҰ¬мҰҳмңјлЎң лҸҷмһ‘н•ңлӢӨ. мҡ”мІӯмқҙ лІ„нӮ·м—җ л“Өм–ҙмҳӨкі , м„Өм •н•ң rateм—җ л§һм¶° мқјм •н•ң мҶҚлҸ„лЎң л№ м ёлӮҳк°„лӢӨ. лІ„нӮ·мқҙ к°Җл“қ м°Ёл©ҙ мғҲ мҡ”мІӯмқҖ лІ„л Өм§„лӢӨ.

    limit_req_dry_run вҖ” м Ғмҡ© м „ н…ҢмҠӨнҠё¶

    мӢӨм„ңл№„мҠӨм—җ Rate Limitingмқ„ мІҳмқҢ м Ғмҡ©н•ҳл©ҙ м •мғҒ нҠёлһҳн”Ҫк№Ңм§Җ м°ЁлӢЁн•ҳлҠ” мӮ¬кі к°Җ лӮңлӢӨ. limit_req_dry_runмқҖ мӢӨм ңлЎң мҡ”мІӯмқ„ кұ°л¶Җн•ҳм§Җ м•Ҡмңјл©ҙм„ң лЎңк·ём—җл§Ң кё°лЎқн•ңлӢӨ. Nginx 1.17.1л¶Җн„° мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

    http {
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
}

server {
    location /api/ {
        limit_req zone=api burst=20 nodelay;
        limit_req_dry_run on;          # мӢӨм ң кұ°л¶Җн•ҳм§Җ м•Ҡкі  лЎңк·ёл§Ң лӮЁк№Җ
        proxy_pass http://backend;
    }
}

    dry run мғҒнғңм—җм„ң $limit_req_status ліҖмҲҳлҘј лЎңк·ём—җ кё°лЎқн•ҳл©ҙ м–ҙл–Ө мҡ”мІӯмқҙ м ңн•ң лҢҖмғҒмқём§Җ нҷ•мқён•  мҲҳ мһҲлӢӨ.

    http {
    log_format dry_run '$remote_addr [$time_local] '
                       '"$request" $status '
                       'limit_req_status=$limit_req_status';

    access_log /var/log/nginx/dryrun.log dry_run;
}

    $limit_req_status к°’:

    к°’ мқҳлҜё
    PASSED м ңн•ң лІ”мң„ лӮҙ вҖ” м •мғҒ нҶөкіј
    DELAYED burst нҒҗм—җм„ң лҢҖкё° мІҳлҰ¬лҗЁ
    REJECTED м ңн•ң мҙҲкіј вҖ” dry_runмқҙ м•„лӢҲм—Ҳмңјл©ҙ кұ°л¶ҖлҗЁ
    DELAYED_DRY_RUN dry run лӘЁл“ңм—җм„ң м§Җм—° лҢҖмғҒ
    REJECTED_DRY_RUN dry run лӘЁл“ңм—җм„ң кұ°л¶Җ лҢҖмғҒ 
    # dry run лЎңк·ём—җм„ң кұ°л¶Җ лҢҖмғҒ нҷ•мқё
grep "REJECTED_DRY_RUN" /var/log/nginx/dryrun.log | \
  awk '{print $1}' | sort | uniq -c | sort -rn | head -20

    мҡҙмҳҒ м Ҳм°Ё: dry_run onмңјлЎң л°°нҸ¬ вҶ’ 1~2мқј лЎңк·ё л¶„м„қ вҶ’ rate/burst к°’ мЎ°м • вҶ’ dry_run offлЎң м „нҷҳ.

    лӢӨмӨ‘ Zone мӨ‘мІ© м Ғмҡ©¶

    н•ҳлӮҳмқҳ locationм—җ м—¬лҹ¬ limit_reqлҘј лҸҷмӢңм—җ м Ғмҡ©н•  мҲҳ мһҲлӢӨ. лӘЁл“  zoneмқҳ мЎ°кұҙмқ„ лҸҷмӢңм—җ л§ҢмЎұн•ҙм•ј мҡ”мІӯмқҙ нҶөкіјн•ңлӢӨ.

    http {
    # IPлӢ№ м ңн•ң
    limit_req_zone $binary_remote_addr zone=per_ip:10m rate=10r/s;

    кіөмң  л©”лӘЁлҰ¬мҷҖ м»Ён…Ңмқҙл„Ҳ мҲҳлӘ…¶

    Nginx м»Ён…Ңмқҙл„Ҳк°Җ мһ¬мӢңмһ‘лҗҳл©ҙ limit_req_zoneмқҳ кіөмң  л©”лӘЁлҰ¬к°Җ мҙҲкё°нҷ”лҗңлӢӨ. Rate Limiting мғҒнғңк°Җ лӮ м•„к°Җл©ҙм„ң мһ¬мӢңмһ‘ м§Ғнӣ„ burst нҠёлһҳн”Ҫмқҙ к·ёлҢҖлЎң л°ұм—”л“ңлЎң м „лӢ¬лҗ  мҲҳ мһҲлӢӨ.

  3. л°°нҸ¬ мӢң rolling updateлЎң м—¬лҹ¬ Podк°Җ мҲңм°Ё мһ¬мӢңмһ‘н•ҳл©ҙ
