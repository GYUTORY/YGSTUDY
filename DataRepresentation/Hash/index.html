    import hashlib

def file_sha256(path: str) -> str:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        while chunk := f.read(8192):
            h.update(chunk)
    return h.hexdigest()

    // Node.js вҖ” Express ETag
const crypto = require('crypto');

function generateETag(body) {
  return crypto
    .createHash('md5')  // ETagлҠ” ліҙм•Ҳ лӘ©м Ғмқҙ м•„лӢҲлқј MD5лЎң м¶©л¶„н•ҳлӢӨ
    .update(body)
    .digest('hex');
}

app.get('/api/products', (req, res) => {
  const data = JSON.stringify(getProducts());
  const etag = generateETag(data);

  if (req.headers['if-none-match'] === etag) {
    return res.status(304).end();
  }

  res.set('ETag', etag);
  res.json(JSON.parse(data));
});

    4. лҚ°мқҙн„° мӨ‘ліө м ңкұ° (Deduplication)¶

    нҢҢмқј м ҖмһҘмҶҢм—җм„ң к°ҷмқҖ нҢҢмқјмқ„ м—¬лҹ¬ лІҲ м ҖмһҘн•ҳм§Җ м•Ҡмңјл Өл©ҙ, нҢҢмқј лӮҙмҡ©мқҳ н•ҙмӢңлҘј нӮӨлЎң м“ҙлӢӨ.

    import hashlib
import os
import shutil

STORE_DIR = "/data/blob-store"

def store_file(source_path: str) -> str:
    """нҢҢмқјмқ„ н•ҙмӢң кё°л°ҳмңјлЎң м ҖмһҘн•ҳкі , н•ҙмӢңк°’мқ„ л°ҳнҷҳн•ңлӢӨ."""
    sha = file_sha256(source_path)
    dest = os.path.join(STORE_DIR, sha[:2], sha[2:4], sha)

    if os.path.exists(dest):
        # мқҙлҜё к°ҷмқҖ лӮҙмҡ©мқҳ нҢҢмқјмқҙ мһҲлӢӨ
        return sha

    os.makedirs(os.path.dirname(dest), exist_ok=True)
    shutil.copy2(source_path, dest)
    return sha

    5. н•ҙмӢң кё°л°ҳ мәҗмӢң нӮӨ¶

    // Java вҖ” мҡ”мІӯ нҢҢлқјлҜён„° мЎ°н•©мңјлЎң мәҗмӢң нӮӨ мғқм„ұ
import java.security.MessageDigest;
import java.nio.charset.StandardCharsets;
import java.util.HexFormat;

public class CacheKeyGenerator {

    public static String generate(String userId, String query, int page) {
        String raw = userId + ":" + query + ":" + page;
        try {
            MessageDigest md = MessageDigest.getInstance("SHA-256");
            byte[] hash = md.digest(raw.getBytes(StandardCharsets.UTF_8));
            return HexFormat.of().formatHex(hash);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}

    мәҗмӢң нӮӨк°Җ л„Ҳл¬ҙ кёёкұ°лӮҳ нҠ№мҲҳл¬ёмһҗк°Җ нҸ¬н•Ёлҗҳл©ҙ Redis к°ҷмқҖ м ҖмһҘмҶҢм—җм„ң л¬ём ңк°Җ мғқкёё мҲҳ мһҲлӢӨ. н•ҙмӢңк°’мқ„ нӮӨлЎң м“°л©ҙ кёёмқҙк°Җ кі м •лҗҳкі  м•Ҳм „н•ң л¬ёмһҗл§Ң нҸ¬н•ЁлҗңлӢӨ.

    н•ҙмӢң м¶©лҸҢ¶

    м¶©лҸҢмқҙлһҖ¶

    м„ңлЎң лӢӨлҘё мһ…л Ҙ A, Bм—җ лҢҖн•ҙ hash(A) == hash(B)мқё кІҪмҡ°лӢӨ. л№„л‘ҳкё°м§‘ мӣҗлҰ¬мғҒ н•ҙмӢң н•ЁмҲҳлҠ” л°ҳл“ңмӢң м¶©лҸҢмқҙ мЎҙмһ¬н•ңлӢӨ. мһ…л Ҙ кіөк°„мқҖ л¬ҙн•ңмқёлҚ° м¶ңл Ҙ кіөк°„мқҖ мң н•ңн•ҳкё° л•Ңл¬ёмқҙлӢӨ.

    Birthday Attack¶

    н•ҙмӢң м¶ңл Ҙмқҙ nл№„нҠёмқҙл©ҙ, м•Ҫ 2^(n/2)к°ңмқҳ мһ…л Ҙл§Ң мӢңлҸ„н•ҳл©ҙ 50% нҷ•лҘ лЎң м¶©лҸҢмқ„ м°ҫмқ„ мҲҳ мһҲлӢӨ. MD5(128л№„нҠё)лҠ” м•Ҫ 2^64лІҲ мӢңлҸ„лЎң м¶©лҸҢмқ„ м°ҫмқ„ мҲҳ мһҲкі , нҳ„лҢҖ н•ҳл“ңмӣЁм–ҙлЎң м¶©л¶„нһҲ к°ҖлҠҘн•ҳлӢӨ. SHA-256(256л№„нҠё)мқҖ 2^128лІҲмқҙ н•„мҡ”н•ҙм„ң нҳ„мӢӨм ҒмңјлЎң л¶Ҳк°ҖлҠҘн•ҳлӢӨ.

    мӢӨл¬ҙм—җм„ң м¶©лҸҢмқҙ л¬ём ңк°Җ лҗҳлҠ” кІҪмҡ°¶

    Gitмқҳ SHA-1 м¶©лҸҢ: 2017л…„ Googleмқҙ SHA-1 м¶©лҸҢмқ„ мӢӨм ңлЎң л§Ңл“Өм–ҙлғҲлӢӨ(SHAttered кіөкІ©). к°ҷмқҖ SHA-1 н•ҙмӢңлҘј к°Җм§„ л‘җ к°ңмқҳ м„ңлЎң лӢӨлҘё PDFлҘј мғқм„ұн–ҲлӢӨ. Gitмқҙ SHA-256мңјлЎң м „нҷҳн•ҳлҠ” мқҙмң лӢӨ.

    н•ҙмӢң н…Ңмқҙлё” DoS: мӣ№ н”„л Ҳмһ„мӣҢнҒ¬м—җм„ң мҡ”мІӯ нҢҢлқјлҜён„°лҘј н•ҙмӢң н…Ңмқҙлё”м—җ л„Јмқ„ л•Ң, к°ҷмқҖ н•ҙмӢң лІ„нӮ·м—җ лӘ°лҰ¬лҸ„лЎқ нҢҢлқјлҜён„°лҘј мЎ°мһ‘н•ҳл©ҙ O(n^2) м„ұлҠҘ м Җн•ҳк°Җ л°ңмғқн•ңлӢӨ. мқҙкұё HashDoSлқјкі  н•ңлӢӨ. Javaмқҳ HashMapмқҖ лІ„нӮ· лӮҙ мҡ”мҶҢк°Җ 8к°ңлҘј л„ҳмңјл©ҙ Red-Black TreeлЎң м „нҷҳн•ҙм„ң мқҙ л¬ём ңлҘј мҷ„нҷ”н•ңлӢӨ.

    мЈјмқҳмӮ¬н•ӯ¶

    • SHA-256мқҙлқјлҸ„ м ҲлҢҖм ҒмңјлЎң м•Ҳм „н•ң кұҙ м•„лӢҲлӢӨ. м–‘мһҗ м»ҙн“Ён„°к°Җ мӢӨмҡ©нҷ”лҗҳл©ҙ Grover м•Ңкі лҰ¬мҰҳмңјлЎң нғҗмғү кіөк°„мқҙ м ңкіұк·јмңјлЎң мӨ„м–ҙл“ лӢӨ. 256л№„нҠёк°Җ 128л№„нҠё мҲҳмӨҖмқҙ лҗңлӢӨлҠ” лң»мқёлҚ°, лӢ№мһҘ кұұм •н•  мҲҳмӨҖмқҖ м•„лӢҲлӢӨ.
    • н•ҙмӢңк°’мқ„ мһҳлқјм„ң(truncate) мӮ¬мҡ©н•ҳл©ҙ м¶©лҸҢ нҷ•лҘ мқҙ кёүкІ©нһҲ мҳ¬лқјк°„лӢӨ. SHA-256 м¶ңл Ҙмқ„ м•һ 8мһҗлҰ¬л§Ң м“°л©ҙ 32л№„нҠё н•ҙмӢңлӮҳ л§Ҳм°¬к°Җм§ҖлӢӨ.
    • л№„л°ҖлІҲнҳё н•ҙмӢұм—җ salt м—Ҷмқҙ мӮ¬мҡ©н•ҳл©ҙ Rainbow table кіөкІ©м—җ л…ём¶ңлҗңлӢӨ. bcrypt/scrypt/Argon2лҠ” saltлҘј мһҗлҸҷмңјлЎң нҸ¬н•Ён•ңлӢӨ.

function fileHash(path, algorithm = 'sha256') {
  return new Promise((resolve, reject) => {
    const hash = crypto.createHash(algorithm);
    const stream = fs.createReadStream(path);
    stream.on('data', chunk => hash.update(chunk));
    stream.on('end', () => resolve(hash.digest('hex')));
    stream.on('error', reject);
  });
}

async function sha256(text) {
