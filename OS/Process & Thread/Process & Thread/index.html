    • HeapмқҖ brk/mmap мӢңмҠӨн…ң мҪңлЎң лҠҳм–ҙлӮңлӢӨ. JVMмқҳ -Xmx к°ҷмқҖ м„Өм •мқҙ мқҙ мҳҒм—ӯ нҒ¬кё°лҘј м ңн•ңн•ҳлҠ” кІғмқҙлӢӨ.
    • Text мҳҒм—ӯмқҖ read-onlyлӢӨ. к°ҷмқҖ л°”мқҙл„ҲлҰ¬лҘј м—¬лҹ¬ н”„лЎңм„ёмҠӨк°Җ мӢӨн–үн•ҳл©ҙ мқҙ мҳҒм—ӯмқҖ л¬јлҰ¬ л©”лӘЁлҰ¬м—җм„ң кіөмң н•ңлӢӨ.

    PCB (Process Control Block)¶

    м»Өл„җмқҙ н”„лЎңм„ёмҠӨлҘј кҙҖлҰ¬н•ҳкё° мң„н•ҙ лӮҙл¶Җм—җ мң м§Җн•ҳлҠ” кө¬мЎ°мІҙлӢӨ. Linuxм—җм„ңлҠ” task_structк°Җ мқҙ м—ӯн• мқ„ н•ңлӢӨ.

    // linux/sched.h (к°„лһөнҷ”)
struct task_struct {
    pid_t pid;                    // н”„лЎңм„ёмҠӨ ID
    volatile long state;          // TASK_RUNNING, TASK_INTERRUPTIBLE л“ұ
    struct mm_struct *mm;         // л©”лӘЁлҰ¬ л§Өн•‘ м •ліҙ (нҺҳмқҙм§Җ н…Ңмқҙлё” л“ұ)

    • D (Uninterruptible Sleep): psм—җм„ң мғҒнғңк°Җ Dмқё н”„лЎңм„ёмҠӨ. NFS л§ҲмҡҙнҠёк°Җ лЁ№нҶөмқҙкұ°лӮҳ л””мҠӨнҒ¬ I/Oк°Җ л§үнһҢ кІҪмҡ°лӢӨ. killлЎңлҸ„ м•Ҳ мЈҪлҠ”лӢӨ. мқҙ мғҒнғңк°Җ мҢ“мқҙл©ҙ м„ңлІ„ м „мІҙк°Җ лҠҗл Өм§„лӢӨ.
    • Z (Zombie): мһҗмӢқ н”„лЎңм„ёмҠӨк°Җ мў…лЈҢлҗҗлҠ”лҚ° л¶ҖлӘЁк°Җ wait()мқ„ м•Ҳ нҳём¶ңн•ң кІҪмҡ°. лҰ¬мҶҢмҠӨлҠ” мқҙлҜё н•ҙм ңлҗҗм§Җл§Ң PID н…Ңмқҙлё”м—җ м—”нҠёлҰ¬к°Җ лӮЁм•„ мһҲлӢӨ. мўҖл№„ мҲҳл°ұ к°ңк°Җ мҢ“мқҙл©ҙ PID кі к°Ҳмқҙ мҳ¬ мҲҳ мһҲлӢӨ.
    • T (Stopped): SIGSTOPмқҙлӮҳ Ctrl+ZлЎң л©Ҳм¶ҳ мғҒнғң. fgлӮҳ SIGCONTлЎң мһ¬к°ңн•ңлӢӨ.

    forkмҷҖ exec вҖ” Linux н”„лЎңм„ёмҠӨ мғқм„ұмқҳ н•өмӢ¬¶

    Linuxм—җм„ң мғҲ н”„лЎңм„ёмҠӨлҘј л§Ңл“ңлҠ” л°©лІ•мқҖ fork() + exec() мЎ°н•©лҝҗмқҙлӢӨ. (м •нҷ•нһҲлҠ” clone() мӢңмҠӨн…ң мҪңмқҙ кё°л°ҳмқҙм§Җл§Ң к°ңл…җм ҒмңјлЎң fork/execлЎң мқҙн•ҙн•ҳл©ҙ лҗңлӢӨ.)

    fork()¶

    нҳ„мһ¬ н”„лЎңм„ёмҠӨлҘј к·ёлҢҖлЎң ліөм ңн•ңлӢӨ. л¶ҖлӘЁмҷҖ мһҗмӢқмқҖ fork м§Ғнӣ„ лҸҷмқјн•ң мҪ”л“ң, лҚ°мқҙн„°, м—ҙлҰ° нҢҢмқј л””мҠӨнҒ¬лҰҪн„°лҘј к°Җм§„лӢӨ.

    #include <unistd.h>
#include <stdio.h>

int main() {
    int x = 10;
    pid_t pid = fork();

    if (pid == 0) {
        // мһҗмӢқ н”„лЎңм„ёмҠӨ
        x = 20;
        printf("child: x=%d, pid=%d\n", x, getpid());
    } else if (pid > 0) {
        // л¶ҖлӘЁ н”„лЎңм„ёмҠӨ
        printf("parent: x=%d, child_pid=%d\n", x, pid);
        wait(NULL); // мһҗмӢқ мў…лЈҢ лҢҖкё° вҖ” м•Ҳ н•ҳл©ҙ мўҖл№„ мғқкёҙлӢӨ
    } else {
        perror("fork failed");
        return 1;
    }
    return 0;
}

    parent: x=10, child_pid=1235
child: x=20, pid=1235

    мһҗмӢқмқҙ x = 20мңјлЎң л°”кҝ”лҸ„ л¶ҖлӘЁмқҳ xлҠ” 10 к·ёлҢҖлЎңлӢӨ. л©”лӘЁлҰ¬ кіөк°„мқҙ л¶„лҰ¬лҗҳм–ҙ мһҲкё° л•Ңл¬ёмқҙлӢӨ. (м •нҷ•нһҲлҠ” CoW л•Ңл¬ём—җ мқҙ мӢңм җм—җ л¶„лҰ¬лҗңлӢӨ.)

    exec()¶

    нҳ„мһ¬ н”„лЎңм„ёмҠӨмқҳ л©”лӘЁлҰ¬лҘј мғҲ н”„лЎңк·ёлһЁмңјлЎң көҗмІҙн•ңлӢӨ. PIDлҠ” к·ёлҢҖлЎң мң м§Җлҗҳкі  мҪ”л“ң, лҚ°мқҙн„°, мҠӨнғқмқҙ м „л¶Җ мғҲ н”„лЎңк·ёлһЁмқҳ кІғмңјлЎң л°”лҖҗлӢӨ.

    #include <unistd.h>
#include <stdio.h>

int main() {
    pid_t pid = fork();

    if (pid == 0) {
        // мһҗмӢқ н”„лЎңм„ёмҠӨм—җм„ң ls мӢӨн–ү
        execl("/bin/ls", "ls", "-la", "/tmp", NULL);
        // exec м„ұкіөн•ҳл©ҙ м—¬кё° лҸ„лӢ¬н•ҳм§Җ м•ҠлҠ”лӢӨ
        perror("exec failed");
        _exit(1);
    } else {
        wait(NULL);
    }
    return 0;
}

    м…ёмқҙ лӘ…л №м–ҙлҘј мӢӨн–үн•ҳлҠ” кіјм •мқҙ м •нҷ•нһҲ мқҙкІғмқҙлӢӨ:

    bash (PID 1000)
  в”ңв”Җ fork()  вҶ’  bash ліөм ңліё (PID 1001)
  в””в”Җ wait()      в””в”Җ exec("/bin/ls") вҶ’ ls (PID 1001) вҶ’ exit

    fork + exec л¶„лҰ¬мқҳ мқҳлҜё¶

    мҷң н•ң лІҲм—җ "мғҲ н”„лЎңм„ёмҠӨлЎң мғҲ н”„лЎңк·ёлһЁ мӢӨн–ү"мқ„ м•Ҳ н•ҳкі  forkмҷҖ execмқ„ л¶„лҰ¬н–Ҳмқ„к№Ң?

    forkмҷҖ exec мӮ¬мқҙм—җ л¶ҖлӘЁ н”„лЎңм„ёмҠӨмқҳ нҷҳкІҪмқ„ мҲҳм •н•  мҲҳ мһҲкё° л•Ңл¬ёмқҙлӢӨ. м…ёмқҳ лҰ¬лӢӨмқҙл үм…ҳ(>, |)мқҙ мқҙ мӣҗлҰ¬лЎң лҸҷмһ‘н•ңлӢӨ.

    pid_t pid = fork();
if (pid == 0) {
    // forkмҷҖ exec мӮ¬мқҙм—җм„ң нҢҢмқј л””мҠӨнҒ¬лҰҪн„° мЎ°мһ‘
    int fd = open("/tmp/output.log", O_WRONLY | O_CREAT, 0644);
    dup2(fd, STDOUT_FILENO);  // stdoutмқ„ нҢҢмқјлЎң лҰ¬лӢӨмқҙл үнҠё
    close(fd);

    execl("/bin/ls", "ls", "-la", NULL);
    _exit(1);
}

    нҢҢмқҙн”„(|)лҸ„ к°ҷмқҖ мӣҗлҰ¬лӢӨ. fork нӣ„ exec м „м—җ pipeмқҳ read/write endлҘј stdin/stdoutм—җ м—°кІ°н•ңлӢӨ.

    Copy-on-Write (CoW)¶

    forkк°Җ н”„лЎңм„ёмҠӨлҘј "ліөм ң"н•ңлӢӨкі  н–ҲлҠ”лҚ°, л©”лӘЁлҰ¬лҘј м „л¶Җ ліөмӮ¬н•ҳл©ҙ лҠҗлҰҙ мҲҳл°–м—җ м—ҶлӢӨ. мӢӨм ңлЎңлҠ” Copy-on-Write л°©мӢқмңјлЎң лҸҷмһ‘н•ңлӢӨ.

    лҸҷмһ‘ мӣҗлҰ¬¶

    1. fork м§Ғнӣ„, л¶ҖлӘЁмҷҖ мһҗмӢқмқҖ к°ҷмқҖ л¬јлҰ¬ нҺҳмқҙм§ҖлҘј к°ҖлҰ¬нӮЁлӢӨ. нҺҳмқҙм§Җ н…Ңмқҙлё”л§Ң ліөмӮ¬н•ҳкі  мӢӨм ң л©”лӘЁлҰ¬лҠ” ліөмӮ¬н•ҳм§Җ м•ҠлҠ”лӢӨ.
    2. лӘЁл“  кіөмң  нҺҳмқҙм§ҖлҘј read-onlyлЎң н‘ңмӢңн•ңлӢӨ.
    3. л¶ҖлӘЁл“  мһҗмӢқмқҙл“  н•ҙлӢ№ нҺҳмқҙм§Җм—җ м“°кё°лҘј мӢңлҸ„н•ҳл©ҙ page faultк°Җ л°ңмғқн•ңлӢӨ.
    4. м»Өл„җмқҙ page faultлҘј л°ӣм•„м„ң к·ё нҺҳмқҙм§Җл§Ң к·ёл•Ң ліөмӮ¬н•ңлӢӨ. м“°кё°лҘј мӢңлҸ„н•ң н”„лЎңм„ёмҠӨм—җ мғҲ л¬јлҰ¬ нҺҳмқҙм§ҖлҘј н• лӢ№н•ҳкі  лӮҙмҡ©мқ„ ліөмӮ¬н•ң л’Ө read-writeлЎң м„Өм •н•ңлӢӨ.
    fork м§Ғнӣ„:
  л¶ҖлӘЁ нҺҳмқҙм§Җ н…Ңмқҙлё” в”Җв”ҖвҶ’ л¬јлҰ¬ нҺҳмқҙм§Җ A (read-only) вҶҗв”Җв”Җ мһҗмӢқ нҺҳмқҙм§Җ н…Ңмқҙлё”

мһҗмӢқмқҙ м“°кё° мӢңлҸ„:
  л¶ҖлӘЁ нҺҳмқҙм§Җ н…Ңмқҙлё” в”Җв”ҖвҶ’ л¬јлҰ¬ нҺҳмқҙм§Җ A (read-write)
  мһҗмӢқ нҺҳмқҙм§Җ н…Ңмқҙлё” в”Җв”ҖвҶ’ л¬јлҰ¬ нҺҳмқҙм§Җ A' (read-write, Aмқҳ ліөмӮ¬ліё)

    CoWк°Җ мӢӨл¬ҙм—җм„ң л¬ём ңлҗҳлҠ” кІҪмҡ°¶

    Redisмқҳ BGSAVEк°Җ лҢҖн‘ңм ҒмқҙлӢӨ. RedisлҠ” мҠӨлғ…мғ·мқ„ лң° л•Ң fork()лҘј м“ҙлӢӨ. fork м§Ғнӣ„м—җлҠ” л©”лӘЁлҰ¬лҘј кұ°мқҳ м¶”к°ҖлЎң м“°м§Җ м•Ҡм§Җл§Ң, м“°кё° мҡ”мІӯмқҙ кі„мҶҚ л“Өм–ҙмҳӨл©ҙ CoWк°Җ л°ңмғқн•ҳл©ҙм„ң л©”лӘЁлҰ¬ мӮ¬мҡ©лҹүмқҙ кёүмҰқн•ңлӢӨ.

    # Redis лЎңк·ём—җм„ң мқҙлҹ° л©”мӢңм§ҖлҘј ліҙл©ҙ CoWлЎң мқён•ң л©”лӘЁлҰ¬ м¶”к°Җ мӮ¬мҡ©лҹүмқҙлӢӨ
# Background saving started by pid 1234
# RDB: 1234 MB of memory used by copy-on-write

    fork нӣ„ л°”лЎң execн•ҳлҠ” кІҪмҡ° (м…ё лӘ…л № мӢӨн–ү л“ұ)м—җлҠ” CoW лҚ•л¶„м—җ л©”лӘЁлҰ¬ ліөмӮ¬к°Җ кұ°мқҳ мқјм–ҙлӮҳм§Җ м•ҠлҠ”лӢӨ. м–ҙм°Ён”ј execмқҙ л©”лӘЁлҰ¬лҘј м „л¶Җ көҗмІҙн•ҳкё° л•Ңл¬ёмқҙлӢӨ.

    Linuxм—җм„ң мҠӨл Ҳл“ңмқҳ мӢӨмІҙ¶

    Linux м»Өл„җмқҖ н”„лЎңм„ёмҠӨмҷҖ мҠӨл Ҳл“ңлҘј кө¬л¶„н•ҳм§Җ м•ҠлҠ”лӢӨ. л‘ҳ лӢӨ task_structлЎң кҙҖлҰ¬н•ңлӢӨ. мҠӨл Ҳл“ңлҠ” clone() мӢңмҠӨн…ң мҪңм—җ CLONE_VM | CLONE_FILES | CLONE_FS | CLONE_SIGHAND н”Ңлһҳк·ёлҘј мЈјм–ҙ мғқм„ұн•ң кІғмқҙлӢӨ. мқҙ н”Ңлһҳк·ёл“Өмқҙ "л©”лӘЁлҰ¬, нҢҢмқј, мӢңк·ёл„җ н•ёл“Өлҹ¬лҘј л¶ҖлӘЁмҷҖ кіөмң н•ҳлқј"лҠ” мқҳлҜёлӢӨ.

    # н”„лЎңм„ёмҠӨмқҳ мҠӨл Ҳл“ң нҷ•мқё

    к·ёлһҳм„ң topм—җм„ң HлҘј лҲ„лҘҙл©ҙ мҠӨл Ҳл“ң лӢЁмң„лЎң CPU мӮ¬мҡ©лҹүмқ„ ліј мҲҳ мһҲлӢӨ. GC мҠӨл Ҳл“ңк°Җ CPUлҘј 100% мһЎм•„лЁ№кі  мһҲлҠ” мғҒнҷ©мқ„ мһЎмқ„ л•Ң мң мҡ©н•ҳлӢӨ.

    л©ҖнӢ° н”„лЎңм„ёмҠӨ vs л©ҖнӢ° мҠӨл Ҳл“ң¶

    м„ңлІ„ м•„нӮӨн…ҚмІҳлҘј м„Өкі„н•  л•Ң мһҗмЈј лӮҳмҳӨлҠ” м„ нғқмқҙлӢӨ.

    л©ҖнӢ° н”„лЎңм„ёмҠӨ л©ҖнӢ° мҠӨл Ҳл“ң
    лҢҖн‘ң мҳҲмӢң nginx worker, PostgreSQL backend Tomcat, Node.js worker_threads
    л©”лӘЁлҰ¬ кІ©лҰ¬ н”„лЎңм„ёмҠӨлі„ лҸ…лҰҪ кіөмң  (н•ң мҠӨл Ҳл“ң лІ„к·ёк°Җ м „мІҙм—җ мҳҒн–Ҙ)
    мғқм„ұ л№„мҡ© лҶ’лӢӨ (CoWлЎң мҷ„нҷ”) лӮ®лӢӨ
    м»Ён…ҚмҠӨнҠё мҠӨмң„м№ӯ л№„мӢёлӢӨ (TLB flush) мғҒлҢҖм ҒмңјлЎң мӢёлӢӨ
    нҶөмӢ  IPC н•„мҡ” (pipe, socket, shm) л©”лӘЁлҰ¬ кіөмң лЎң л°”лЎң к°ҖлҠҘ
    м•Ҳм •м„ұ н•ң н”„лЎңм„ёмҠӨ мЈҪм–ҙлҸ„ лӢӨлҘё н”„лЎңм„ёмҠӨ л¬ҙкҙҖ н•ң мҠӨл Ҳл“ңк°Җ SIGSEGV л°ӣмңјл©ҙ н”„лЎңм„ёмҠӨ м „мІҙ мӮ¬л§қ

    nginxк°Җ л©ҖнӢ° н”„лЎңм„ёмҠӨ лӘЁлҚёмқ„ м“°лҠ” мқҙмң лҠ”, worker н•ҳлӮҳк°Җ мЈҪм–ҙлҸ„ masterк°Җ мғҲлЎң forkн•ҳл©ҙ лҗҳкё° л•Ңл¬ёмқҙлӢӨ. JVMмқҖ л©ҖнӢ° мҠӨл Ҳл“ң лӘЁлҚёмқёлҚ°, GCлҘј нҸ¬н•Ён•ң лӮҙл¶Җ лҸҷмһ‘мқҙ мҠӨл Ҳл“ң кё°л°ҳмңјлЎң м„Өкі„лҗҳм–ҙ мһҲлӢӨ.

    мҠӨл Ҳл“ң лҸҷкё°нҷ” вҖ” мҷң н•„мҡ”н•ҳкі  лӯҳ м“°лҠ”к°Җ¶

    м—¬лҹ¬ мҠӨл Ҳл“ңк°Җ к°ҷмқҖ ліҖмҲҳлҘј мқҪкі  м“°л©ҙ л¬ём ңк°Җ мғқкёҙлӢӨ.

    // мқҙ мҪ”л“ңлҠ” к№Ём§„лӢӨ
private int count = 0;

// мҠӨл Ҳл“ң A, Bк°Җ лҸҷмӢңм—җ мӢӨн–ү
public void increment() {
    count++;  // мқҪкё° вҶ’ лҚ”н•ҳкё° вҶ’ м“°кё° 3лӢЁкі„. мӣҗмһҗм Ғмқҙм§Җ м•ҠлӢӨ.
}

    count++лҠ” н•ҳлӮҳмқҳ м—°мӮ°мІҳлҹј ліҙмқҙм§Җл§Ң мӢӨм ңлЎңлҠ” LOAD вҶ’ ADD вҶ’ STORE 3к°ң лӘ…л №м–ҙлӢӨ. л‘җ мҠӨл Ҳл“ңк°Җ лҸҷмӢңм—җ LOADн•ҳл©ҙ н•ҳлӮҳмқҳ мҰқк°Җл¶„мқҙ мӮ¬лқјм§„лӢӨ.

    мЈјмҡ” лҸҷкё°нҷ” лҸ„кө¬¶

    Mutex (Mutual Exclusion): н•ң лІҲм—җ н•ҳлӮҳмқҳ мҠӨл Ҳл“ңл§Ң мһ„кі„ мҳҒм—ӯм—җ м§„мһ…н•ңлӢӨ.

    pthread_mutex_t lock = PTHREAD_MUTEX_INITIALIZER;

void increment() {
    pthread_mutex_lock(&lock);
    count++;
    pthread_mutex_unlock(&lock);
}

    Semaphore: лҸҷмӢңм—җ Nк°ң мҠӨл Ҳл“ңк№Ңм§Җ м ‘к·ј н—Ҳмҡ©. DB м»Өл„Ҙм…ҳ н’ҖмІҳлҹј м ңн•ңлҗң лҰ¬мҶҢмҠӨлҘј кҙҖлҰ¬н•  л•Ң м“ҙлӢӨ.

    Read-Write Lock: мқҪкё°лҠ” м—¬лҹ¬ мҠӨл Ҳл“ңк°Җ лҸҷмӢңм—җ, м“°кё°лҠ” лӢЁлҸ…мңјлЎң. мқҪкё°к°Җ л§Һкі  м“°кё°к°Җ л“ңл¬ё м„Өм •к°’ мәҗмӢң к°ҷмқҖ кіім—җ м“ҙлӢӨ.

    Atomic м—°мӮ°: лқҪ м—Ҷмқҙ лӢЁмқј ліҖмҲҳмқҳ мӣҗмһҗм Ғ мЎ°мһ‘. count++ м •лҸ„лҠ” AtomicIntegerлӮҳ __atomic_fetch_addлЎң мІҳлҰ¬н•ҳлҠ” кІҢ лқҪліҙлӢӨ л№ лҘҙлӢӨ.

    лҚ°л“ңлқҪ¶

    л‘җ мҠӨл Ҳл“ңк°Җ м„ңлЎң мғҒлҢҖк°Җ к°Җм§„ лқҪмқ„ кё°лӢӨлҰ¬л©ҙ мҳҒмӣҗнһҲ л©Ҳм¶ҳлӢӨ.

    Thread A: lock(mutex1) вҶ’ lock(mutex2)  // mutex2лҘј кё°лӢӨлҰј
Thread B: lock(mutex2) вҶ’ lock(mutex1)  // mutex1мқ„ кё°лӢӨлҰј

    л°©м§Җ л°©лІ•мқҖ лӢЁмҲңн•ҳлӢӨ: лқҪ нҡҚл“қ мҲңм„ңлҘј н•ӯмғҒ к°ҷкІҢ мң м§Җн•ңлӢӨ. лӘЁл“  мҪ”л“ңм—җм„ң mutex1 вҶ’ mutex2 мҲңм„ңлЎң мһЎмңјл©ҙ лҚ°л“ңлқҪмқҙ л°ңмғқн•ҳм§Җ м•ҠлҠ”лӢӨ.

    м„ңлІ„к°Җ к°‘мһҗкё° мқ‘лӢөмқ„ м•Ҳ н•ҳл©ҙм„ң CPU мӮ¬мҡ©лҘ мқҙ 0%м—җ к°Җк№Ңмҡ°л©ҙ лҚ°л“ңлқҪмқ„ мқҳмӢ¬н•ҙм•ј н•ңлӢӨ. JVMмқҙл©ҙ jstackмңјлЎң мҠӨл Ҳл“ң лҚӨн”„лҘј лң¬лӢӨ.

    $ jstack <pid> | grep -A 5 "deadlock"
