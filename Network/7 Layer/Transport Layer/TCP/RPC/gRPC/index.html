    // мқёмҰқ Interceptor
public class AuthInterceptor implements ServerInterceptor {

    @Override
    public <ReqT, RespT> ServerCall.Listener<ReqT> interceptCall(
            ServerCall<ReqT, RespT> call,
            Metadata headers,
            ServerCallHandler<ReqT, RespT> next) {

        String token = headers.get(
            Metadata.Key.of("authorization", Metadata.ASCII_STRING_MARSHALLER));

        if (token == null || !validateToken(token)) {
            call.close(Status.UNAUTHENTICATED
                .withDescription("мң нҡЁн•ҳм§Җ м•ҠмқҖ мқёмҰқ нҶ нҒ°"), new Metadata());
            return new ServerCall.Listener<>() {};  // л№Ҳ лҰ¬мҠӨл„Ҳ л°ҳнҷҳ
        }

        // мқёмҰқ м„ұкіө мӢң мӮ¬мҡ©мһҗ м •ліҙлҘј Contextм—җ м ҖмһҘ
        Context ctx = Context.current()
            .withValue(USER_ID_KEY, extractUserId(token));

        return Contexts.interceptCall(ctx, call, headers, next);
    }

    // Context Key м •мқҳ
    public static final Context.Key<String> USER_ID_KEY = Context.key("userId");
}

public class LoggingInterceptor implements ServerInterceptor {

    @Override
    public <ReqT, RespT> ServerCall.Listener<ReqT> interceptCall(
            ServerCall<ReqT, RespT> call,
            Metadata headers,
            ServerCallHandler<ReqT, RespT> next) {

        String method = call.getMethodDescriptor().getFullMethodName();
        long startTime = System.currentTimeMillis();
        log.info("gRPC нҳём¶ң мӢңмһ‘: {}", method);

        // мқ‘лӢө мӢң лЎңк№…мқ„ мң„н•ҙ ServerCallмқ„ лһҳн•‘
        ServerCall<ReqT, RespT> wrappedCall = new ForwardingServerCall
                .SimpleForwardingServerCall<>(call) {
            @Override
            public void close(Status status, Metadata trailers) {
                long elapsed = System.currentTimeMillis() - startTime;
                log.info("gRPC нҳём¶ң мҷ„лЈҢ: {} | status={} | {}ms",
                    method, status.getCode(), elapsed);
                super.close(status, trailers);
            }
        };

        return next.startCall(wrappedCall, headers);
    }
}

    // л©”нҠёлҰӯ мҲҳм§‘ Interceptor
public class MetricsInterceptor implements ServerInterceptor {

    private final MeterRegistry meterRegistry;

    @Override
    public <ReqT, RespT> ServerCall.Listener<ReqT> interceptCall(
            ServerCall<ReqT, RespT> call,
            Metadata headers,
            ServerCallHandler<ReqT, RespT> next) {

        String method = call.getMethodDescriptor().getFullMethodName();
        Timer.Sample sample = Timer.start(meterRegistry);

        ServerCall<ReqT, RespT> wrappedCall = new ForwardingServerCall
                .SimpleForwardingServerCall<>(call) {
            @Override
            public void close(Status status, Metadata trailers) {
                sample.stop(Timer.builder("grpc.server.calls")
                    .tag("method", method)
                    .tag("status", status.getCode().name())
                    .register(meterRegistry));
                meterRegistry.counter("grpc.server.calls.total",
                    "method", method,
                    "status", status.getCode().name()).increment();
                super.close(status, trailers);
            }
        };

        return next.startCall(wrappedCall, headers);
    }
}

    @GrpcService
public class UserGrpcService extends UserServiceGrpc.UserServiceImplBase {

    private final UserRepository userRepository;

    public UserGrpcService(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    @Override
    public void getUser(GetUserRequest request,
                        StreamObserver<GetUserResponse> responseObserver) {
        User user = userRepository.findById(request.getUserId())
            .orElse(null);

        if (user == null) {
            responseObserver.onError(Status.NOT_FOUND
                .withDescription("мӮ¬мҡ©мһҗлҘј м°ҫмқ„ мҲҳ м—ҶмқҢ: " + request.getUserId())
                .asRuntimeException());
            return;
        }

        GetUserResponse response = GetUserResponse.newBuilder()
            .setUserId(user.getId())
            .setName(user.getName())
            .setEmail(user.getEmail())
            .setRole(UserRole.valueOf(user.getRole()))
            .build();

        responseObserver.onNext(response);
        responseObserver.onCompleted();
    }

    @Override
    public void listUsers(ListUsersRequest request,
                          StreamObserver<GetUserResponse> responseObserver) {
        Page<User> users = userRepository.findAll(
            PageRequest.of(request.getPage(), request.getSize()));

        for (User user : users.getContent()) {
            GetUserResponse response = GetUserResponse.newBuilder()
                .setUserId(user.getId())
                .setName(user.getName())
                .setEmail(user.getEmail())
                .build();
            responseObserver.onNext(response);
        }
        responseObserver.onCompleted();
    }

    @Override
    public void createUser(CreateUserRequest request,
                           StreamObserver<GetUserResponse> responseObserver) {
        // мһ…л Ҙ кІҖмҰқ
        if (request.getName().isBlank()) {
            responseObserver.onError(Status.INVALID_ARGUMENT
                .withDescription("nameмқҖ л№„м–ҙмһҲмқ„ мҲҳ м—ҶмқҢ")
                .asRuntimeException());
            return;
        }

        User user = new User();
        user.setName(request.getName());
        user.setEmail(request.getEmail());
        user.setRole(request.getRole().name());
        userRepository.save(user);

        GetUserResponse response = GetUserResponse.newBuilder()
            .setUserId(user.getId())
            .setName(user.getName())
            .setEmail(user.getEmail())
            .setRole(request.getRole())
            .build();

        responseObserver.onNext(response);
        responseObserver.onCompleted();
    }
}

                    throw new ServiceUnavailableException("user-service");

                    throw new GrpcCallException(e);
            }
        }

    • maxConnectionAgeGrace вҖ” GOAWAYлҘј ліҙлӮё нӣ„ м§„н–ү мӨ‘мқё RPCк°Җ мҷ„лЈҢлҗ  л•Ңк№Ңм§Җ кё°лӢӨлҰ¬лҠ” мӢңк°„.

    м»Өл„Ҙм…ҳ кҙҖлҰ¬ мЈјмқҳмӮ¬н•ӯ¶

    gRPC мұ„л„җ(ManagedChannel)мқҖ м• н”ҢлҰ¬мјҖмқҙм…ҳ мӢңмһ‘ мӢң н•ң лІҲ мғқм„ұн•ҳкі  мһ¬мӮ¬мҡ©н•ҙм•ј н•ңлӢӨ. нҳём¶ңл§ҲлӢӨ мғҲ мұ„л„җмқ„ л§Ңл“Өл©ҙ TCP м»Өл„Ҙм…ҳмқҙ кі„мҶҚ м—ҙлҰ¬кі  лӢ«нһҲл©ҙм„ң м„ұлҠҘмқҙ кёүкІ©нһҲ л–Ём–ҙм§„лӢӨ. мұ„л„җ лӮҙл¶Җм—җ м»Өл„Ҙм…ҳ н’Җмқҙ мһҲмңјлҜҖлЎң м§Ғм ‘ н’Җмқ„ кө¬нҳ„н•  н•„мҡ” м—ҶлӢӨ.

    // мһҳлӘ»лҗң мӮ¬мҡ© вҖ” л§Ө нҳём¶ңл§ҲлӢӨ мұ„л„җ мғқм„ұ

    лЎңл“ңл°ёлҹ°мӢұ¶

    gRPCлҠ” HTTP/2мқҳ л©ҖнӢ°н”Ңл үмӢұ л•Ңл¬ём—җ лЎңл“ңл°ёлҹ°мӢұмқҙ RESTмҷҖ лӢӨлҘҙкІҢ лҸҷмһ‘н•ңлӢӨ. HTTP/1.1м—җм„ңлҠ” мҡ”мІӯл§ҲлӢӨ м»Өл„Ҙм…ҳмқ„ л§әкұ°лӮҳ, м»Өл„Ҙм…ҳмқ„ мһ¬мӮ¬мҡ©н•ҳлҚ”лқјлҸ„ н•ң м»Өл„Ҙм…ҳм—җ н•ң мҡ”мІӯм”© мІҳлҰ¬н•ңлӢӨ. L4 лЎңл“ңл°ёлҹ°м„ңк°Җ м»Өл„Ҙм…ҳ лӢЁмң„лЎң л¶„л°°н•ҳл©ҙ мһҗм—°мҠӨлҹҪкІҢ мҡ”мІӯмқҙ л¶„мӮ°лҗңлӢӨ.

    gRPCлҠ” н•ҳлӮҳмқҳ м»Өл„Ҙм…ҳмңјлЎң лӘЁл“  мҡ”мІӯмқ„ ліҙлӮёлӢӨ. L4 лЎңл“ңл°ёлҹ°м„ңлҠ” м»Өл„Ҙм…ҳмқҙ л§әм–ҙм§Ҳ л•Ң н•ң лІҲл§Ң л°ұм—”л“ңлҘј м„ нғқн•ҳлҜҖлЎң, лӘЁл“  мҡ”мІӯмқҙ н•ң м„ңлІ„лЎң лӘ°лҰ°лӢӨ.

    н”„лЎқмӢң кё°л°ҳ лЎңл“ңл°ёлҹ°мӢұ (L7)¶

    Envoy, nginx(gRPC м§Җмӣҗ лІ„м „), Istio к°ҷмқҖ L7 н”„лЎқмӢңлҘј мӮ¬мҡ©н•ңлӢӨ. н”„лЎқмӢңк°Җ HTTP/2 н”„л Ҳмһ„ лӢЁмң„лЎң мҡ”мІӯмқ„ нҢҢмӢұн•ҙм„ң к°ңлі„ RPC нҳём¶ңмқ„ лӢӨлҘё л°ұм—”л“ңлЎң л¶„л°°н•ңлӢӨ.

    Client вҶ’ L7 Proxy (Envoy) вҶ’ Backend A
                           вҶ’ Backend B
                           вҶ’ Backend C

    нҒҙлқјмқҙм–ёнҠёлҠ” н”„лЎқмӢң мЈјмҶҢл§Ң м•Ңл©ҙ лҗңлӢӨ. л°ұм—”л“ң м¶”к°Җ/м ңкұ°лҘј н”„лЎқмӢңм—җм„ң мІҳлҰ¬н•ҳлҜҖлЎң нҒҙлқјмқҙм–ёнҠё ліҖкІҪмқҙ м—ҶлӢӨ. лӢӨл§Ң н”„лЎқмӢңк°Җ лі‘лӘ©мқҙ лҗ  мҲҳ мһҲкі , м¶”к°Җ hopмңјлЎң м§Җм—°мқҙ мғқкёҙлӢӨ.

    нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң лЎңл“ңл°ёлҹ°мӢұ¶

    нҒҙлқјмқҙм–ёнҠёк°Җ м§Ғм ‘ м—¬лҹ¬ м„ңлІ„ мЈјмҶҢлҘј м•Ңкі , мҡ”мІӯл§ҲлӢӨ лӢӨлҘё м„ңлІ„лҘј м„ нғқн•ңлӢӨ. н”„лЎқмӢң м—Ҷмқҙ м§Ғм ‘ нҶөмӢ н•ҳлҜҖлЎң м§Җм—°мқҙ м ҒлӢӨ.

    gRPCлҠ” round_robin, pick_first к°ҷмқҖ лӮҙмһҘ м •мұ…мқ„ м ңкіөн•ңлӢӨ.

    // нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң round-robin
ManagedChannel channel = ManagedChannelBuilder
    .forTarget("dns:///my-grpc-service.default.svc.cluster.local:50051")
    .defaultLoadBalancingPolicy("round_robin")
    .usePlaintext()
    .build();

    dns:/// нҳ•мӢқмқ„ м“°л©ҙ DNSм—җм„ң A л ҲмҪ”л“ңлҘј мЎ°нҡҢн•ҙм„ң м—¬лҹ¬ IPлҘј к°Җм ёмҳӨкі , round-robinмңјлЎң л¶„л°°н•ңлӢӨ.

    Kubernetes нҷҳкІҪм—җм„ң мЈјмқҳн•  м җ¶

    K8sмқҳ кё°ліё ServiceлҠ” ClusterIP нғҖмһ…мңјлЎң, kube-proxyк°Җ L4 лЎңл“ңл°ёлҹ°мӢұмқ„ н•ңлӢӨ. gRPCмҷҖ L4 мЎ°н•©мқҖ мң„м—җм„ң м„ӨлӘ…н•ң лҢҖлЎң нҠёлһҳн”Ҫ мҸ лҰјмқҙ л°ңмғқн•ңлӢӨ.

    н•ҙкІ° л°©лІ•мқҖ нҒ¬кІҢ м„ё к°Җм§ҖлӢӨ:

    1. Headless Service + нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң лЎңл“ңл°ёлҹ°мӢұ

    # K8s Headless Service
apiVersion: v1
kind: Service
metadata:
  name: user-service
spec:
  clusterIP: None    # Headless вҖ” DNSк°Җ лӘЁл“  Pod IPлҘј л°ҳнҷҳ
  selector:
    app: user-service
  ports:
    - port: 50051

    Headless ServiceлҠ” DNS мЎ°нҡҢ мӢң лӘЁл“  Podмқҳ IPлҘј л°ҳнҷҳн•ңлӢӨ. нҒҙлқјмқҙм–ёнҠёк°Җ round-robinмңјлЎң мҡ”мІӯмқ„ л¶„л°°н•ңлӢӨ.

    л¬ём ңлҠ” DNS мәҗмӢұмқҙлӢӨ. Podк°Җ м¶”к°Җ/мӮӯм ңлҗҳм–ҙлҸ„ DNS мәҗмӢң л§ҢлЈҢ м „к№Ңм§ҖлҠ” мқҙм „ лӘ©лЎқмқ„ мӮ¬мҡ©н•ңлӢӨ. maxConnectionAgeлҘј м„Өм •н•ҙм„ң мЈјкё°м ҒмңјлЎң м»Өл„Ҙм…ҳмқ„ к°ұмӢ н•ҳл©ҙ мғҲ Pod лӘ©лЎқмқ„ л°ҳмҳҒн•  мҲҳ мһҲлӢӨ.

    2. Istio/Envoy мӮ¬мқҙл“ңм№ҙ

    м„ңл№„мҠӨ л©”мӢңлҘј мӮ¬мҡ©н•ҳл©ҙ к°Ғ Pod мҳҶм—җ Envoy н”„лЎқмӢңк°Җ л¶ҷм–ҙм„ң L7 лЎңл“ңл°ёлҹ°мӢұмқ„ мһҗлҸҷмңјлЎң мІҳлҰ¬н•ңлӢӨ. м• н”ҢлҰ¬мјҖмқҙм…ҳ мҪ”л“ң ліҖкІҪ м—Ҷмқҙ лҸҷмһ‘н•ңлӢӨ.

    3. gRPCмҡ© L7 Ingress

    grpc-web н”„лЎңнҶ мҪңмқҙлӮҳ nginx ingressмқҳ gRPC л°ұм—”л“ң м„Өм •мқ„ мӮ¬мҡ©н•ңлӢӨ.

    мӢӨл¬ҙм—җм„ңлҠ” мқҙлҜё Istio к°ҷмқҖ м„ңл№„мҠӨ л©”мӢңлҘј мӮ¬мҡ© мӨ‘мқҙл©ҙ мӮ¬мқҙл“ңм№ҙ л°©мӢқмқҙ нҺён•ҳкі , к·ёл Үм§Җ м•Ҡмңјл©ҙ Headless Service + нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң лЎңл“ңл°ёлҹ°мӢұмқҙ к°ҖмһҘ лӢЁмҲңн•ҳлӢӨ.

    л””лІ„к№… лҸ„кө¬¶

    gRPCлҠ” л°”мқҙл„ҲлҰ¬ н”„лЎңнҶ мҪңмқҙлқј curlлЎң н…ҢмҠӨнҠён•  мҲҳ м—ҶлӢӨ. м „мҡ© CLI лҸ„кө¬к°Җ н•„мҡ”н•ҳлӢӨ.

    grpcurl¶

    RESTм—җм„ң curlмқҙ н•ҳлҠ” м—ӯн• мқ„ gRPCм—җм„ң н•ңлӢӨ. м„ңлІ„мқҳ лҰ¬н”Ңл үм…ҳ APIлҘј нҶөн•ҙ м„ңл№„мҠӨ лӘ©лЎқкіј л©”м„ңл“ңлҘј мЎ°нҡҢн•ҳкі , мҡ”мІӯмқ„ ліҙлӮј мҲҳ мһҲлӢӨ.

    # м„Өм№ҳ (macOS)

    мқён„°лһҷнӢ°лёҢ gRPC нҒҙлқјмқҙм–ёнҠёлӢӨ. REPL лӘЁл“ңм—җм„ң нғӯ мҷ„м„ұмқ„ м§Җмӣҗн•ҙм„ң proto кө¬мЎ°лҘј нғҗмғүн•ҳл©ҙм„ң мҡ”мІӯмқ„ ліҙлӮҙкё° нҺён•ҳлӢӨ.

    мқёмҰқ нҶ нҒ° мІҳлҰ¬¶

    мқёмҰқмқҖ Interceptor м„№м…ҳм—җм„ң м„ӨлӘ…н•ң нҢЁн„ҙмқ„ мӮ¬мҡ©н•ңлӢӨ. л©”нғҖлҚ°мқҙн„°м—җ authorization н—ӨлҚ”лҘј л„Јкі , м„ңлІ„ Interceptorм—җм„ң кІҖмҰқн•ҳлҠ” л°©мӢқмқҙлӢӨ.

    мҡ”м•Ҫ¶

    gRPCлҠ” л§ҲмқҙнҒ¬лЎңм„ңл№„мҠӨ к°„ лӮҙл¶Җ нҶөмӢ м—җ м Ғн•©н•ҳлӢӨ. HTTP/2 л©ҖнӢ°н”Ңл үмӢұмңјлЎң н•ҳлӮҳмқҳ м»Өл„Ҙм…ҳм—җ м—¬лҹ¬ мҡ”мІӯмқ„ мІҳлҰ¬н•ҳкі , Protocol BuffersлЎң лҚ°мқҙн„° нҒ¬кё°лҘј мӨ„мқёлӢӨ.

    мӢӨл¬ҙм—җм„ң мӢ кІҪ мҚЁм•ј н•  кІғл“Ө:

    • deadlineмқҖ л°ҳл“ңмӢң м„Өм •н•ңлӢӨ вҖ” м•Ҳ н•ҳл©ҙ cascading failure мң„н—ҳ
    • proto н•„л“ң лІҲнҳёлҠ” мӮӯм ң/мһ¬мӮ¬мҡ©н•ҳм§Җ м•ҠлҠ”лӢӨ вҖ” reserved мӮ¬мҡ©
    • L4 лЎңл“ңл°ёлҹ°м„ңлЎңлҠ” нҠёлһҳн”Ҫмқҙ л¶„мӮ°лҗҳм§Җ м•ҠлҠ”лӢӨ вҖ” L7 н”„лЎқмӢң лҳҗлҠ” нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң лЎңл“ңл°ёлҹ°мӢұ н•„мҡ”
    • keepaliveлҘј м„Өм •н•ңлӢӨ вҖ” мӨ‘к°„ мһҘл№„к°Җ мң нңҙ м»Өл„Ҙм…ҳмқ„ лҒҠлҠ”лӢӨ
    • ManagedChannelмқҖ мһ¬мӮ¬мҡ©н•ңлӢӨ вҖ” нҳём¶ңл§ҲлӢӨ мғқм„ұн•ҳл©ҙ м„ұлҠҘ м Җн•ҳ

    gRPCлҘј м“°кё° м Ғн•©н•ң кІҪмҡ°¶
