    // мқҙ мӢңм җм—җм„ң DB, мәҗмӢң лӘЁл‘җ мҙҲкё°нҷ”к°Җ лҒқлӮң мғҒнғң
    const dbHealthy = await this.dbService.ping();
    const cacheHealthy = await this.cacheService.ping();

    if (!dbHealthy || !cacheHealthy) {
      throw new Error('н•„мҲҳ мқён”„лқј м—°кІ° мӢӨнҢЁ, м„ңлІ„лҘј мӢңмһ‘н•  мҲҳ м—ҶмқҢ');
    }
  }
}

    OnModuleDestroy¶

    app.close()к°Җ нҳём¶ңлҗҳкұ°лӮҳ н”„лЎңм„ёмҠӨ мў…лЈҢ мӢңк·ёл„җмқ„ л°ӣмңјл©ҙ мӢӨн–үлҗңлӢӨ. лҰ¬мҶҢмҠӨ м •лҰ¬ мһ‘м—…(DB м»Өл„Ҙм…ҳ н’Җ мў…лЈҢ, м—ҙлҰ° нҢҢмқј лӢ«кё° л“ұ)мқ„ м—¬кё°м„ң н•ңлӢӨ.

    @Injectable()
export class DatabaseService implements OnModuleDestroy {
  async onModuleDestroy() {
    await this.connectionPool.end();
    console.log('DB м»Өл„Ҙм…ҳ н’Җ мў…лЈҢ');
  }
}

    BeforeApplicationShutdown¶

    OnModuleDestroyк°Җ лӘЁл‘җ лҒқлӮң нӣ„ мӢӨн–үлҗңлӢӨ. мқҙ нӣ…м—җлҠ” мӢңк·ёл„җ л¬ёмһҗм—ҙмқҙ нҢҢлқјлҜён„°лЎң л“Өм–ҙмҳЁлӢӨ. HTTP м„ңлІ„ м—°кІ°мқҙ м•„м§Ғ м—ҙл ӨмһҲлҠ” мӢңм җмқҙлқј м§„н–ү мӨ‘мқё мҡ”мІӯмқ„ л§Ҳл¬ҙлҰ¬н•  мҲҳ мһҲлӢӨ.

    @Injectable()
export class ShutdownService implements BeforeApplicationShutdown {
  beforeApplicationShutdown(signal?: string) {
    console.log(`мў…лЈҢ мӢңк·ёл„җ: ${signal}`);  // 'SIGTERM', 'SIGINT' л“ұ
    // мқҙ мӢңм җм—җм„ң HTTP м—°кІ°мқҖ м•„м§Ғ мӮҙм•„мһҲмқҢ
    // м§„н–ү мӨ‘мқё мҡ”мІӯмқҙ мҷ„лЈҢлҗ  л•Ңк№Ңм§Җ лҢҖкё°н•ҳлҠ” лЎңм§Ғмқ„ л„Јмқ„ мҲҳ мһҲмқҢ
  }
}

    мӢӨл¬ҙм—җм„ң нқ”нһҲ мһҳлӘ» м“°лҠ” мӮ¬лЎҖ¶

    1. OnModuleInitм—җм„ң лӢӨлҘё лӘЁл“Ҳмқҳ мҙҲкё°нҷ” кІ°кіјм—җ мқҳмЎҙн•ҳлҠ” кІҪмҡ°

@Injectable()

  constructor(private readonly templateService: TemplateService) {}

    // NotificationModuleмқҙ TemplateModuleліҙлӢӨ лЁјм Җ мҙҲкё°нҷ”лҗҳл©ҙ

    this.defaultTemplate = this.templateService.getTemplate('welcome');

    мқҙлҹ° кІҪмҡ°лҠ” OnApplicationBootstrapм—җ л„Јм–ҙм•ј н•ңлӢӨ. OnApplicationBootstrapмқҖ лӘЁл“  лӘЁл“Ҳмқҳ OnModuleInitмқҙ лҒқлӮң нӣ„ мӢӨн–үлҗҳлӢҲк№Ң.

    2. мғқм„ұмһҗм—җ л№„лҸҷкё° мҙҲкё°нҷ” лЎңм§Ғмқ„ л„ЈлҠ” кІҪмҡ°

    3. OnModuleDestroyм—җм„ң мў…лЈҢ мҲңм„ңлҘј кі л Өн•ҳм§Җ м•ҠлҠ” кІҪмҡ°

    // к·јлҚ° DatabaseServiceк°Җ мқҙлҜё destroyлҗҳм–ҙ м»Өл„Ҙм…ҳмқҙ лҒҠкІјмңјл©ҙ?
  }

    OnModuleDestroyмқҳ мӢӨн–ү мҲңм„ңлҠ” OnModuleInitмқҳ м—ӯмҲңмқҙлӢӨ. imports л°°м—ҙ мҲңм„ңк°Җ [DatabaseModule, LogModule]мқҙл©ҙ, destroyлҠ” LogModule вҶ’ DatabaseModule мҲңмқҙлӢӨ. мҲңм„ңлҘј м ңм–ҙн•ҳл Өл©ҙ imports л°°м—ҙ мҲңм„ңлҘј мӢ кІҪ мҚЁм•ј н•ңлӢӨ.

    лҸҷм Ғ лӘЁл“Ҳ нҢЁн„ҙ¶

    м •м Ғ лӘЁл“ҲмқҖ @Module() лҚ°мҪ”л Ҳмқҙн„°м—җ лӘЁл“  кІғмқҙ кі м •лҗҳм–ҙ мһҲлӢӨ. лҸҷм Ғ лӘЁл“ҲмқҖ лҹ°нғҖмһ„м—җ м„Өм •мқ„ л°ӣм•„м„ң лӘЁл“Ҳмқҳ н”„лЎңл°”мқҙлҚ” кө¬м„ұмқ„ л°”кҝҖ мҲҳ мһҲлӢӨ.

    forRoot / forRootAsync¶

    м „м—ӯ м„Өм •мқҙ н•„мҡ”н•ң лӘЁл“Ҳм—җ мӮ¬мҡ©н•ңлӢӨ. м• н”ҢлҰ¬мјҖмқҙм…ҳ м „мІҙм—җм„ң н•ҳлӮҳмқҳ м„Өм •мңјлЎң кіөмң н•ҙм•ј н•  л•Ң м“ҙлӢӨ.

    @Module({})
export class DatabaseModule {
  static forRoot(options: DatabaseOptions): DynamicModule {
    return {
      module: DatabaseModule,
      global: true,  // н•ң лІҲ л“ұлЎқн•ҳл©ҙ м–ҙл””м„ңл“  import м—Ҷмқҙ мӮ¬мҡ© к°ҖлҠҘ
      providers: [
        {
          provide: 'DATABASE_OPTIONS',
          useValue: options,
        },
        DatabaseService,
      ],
      exports: [DatabaseService],
    };
  }
}

    лӮҙл¶Җм ҒмңјлЎң мқјм–ҙлӮҳлҠ” мқј: forRoot()лҠ” лӢЁмҲңнһҲ DynamicModule к°қмІҙлҘј лҰ¬н„ҙн•ҳлҠ” м •м Ғ л©”м„ңл“ңлӢӨ. NestJSлҠ” мқҙ лҰ¬н„ҙк°’мқ„ л°ӣм•„м„ң мқјл°ҳ лӘЁл“Ҳмқҳ л©”нғҖлҚ°мқҙн„°мІҳлҹј мІҳлҰ¬н•ңлӢӨ. module н•„л“ңм—җ м§Җм •лҗң нҒҙлһҳмҠӨк°Җ лӘЁл“Ҳ нҶ нҒ°мқҙ лҗҳкі , providersмҷҖ exportsк°Җ к·ё лӘЁл“Ҳмқҳ л©”нғҖлҚ°мқҙн„°м—җ н•©міҗм§„лӢӨ.

    // мӮ¬мҡ©н•ҳлҠ” мӘҪ
@Module({
  imports: [
    DatabaseModule.forRoot({
      host: 'localhost',
      port: 5432,
      database: 'myapp',
    }),
  ],
})
export class AppModule {}

    forRootAsyncлҠ” м„Өм •к°’мқ„ л№„лҸҷкё°лЎң к°Җм ёмҷҖм•ј н•  л•Ң мӮ¬мҡ©н•ңлӢӨ. ConfigServiceм—җм„ң нҷҳкІҪліҖмҲҳлҘј мқҪм–ҙм•ј н•ҳлҠ” кІҪмҡ°к°Җ лҢҖн‘ңм ҒмқҙлӢӨ.

    @Module({})
export class DatabaseModule {
  static forRootAsync(options: DatabaseAsyncOptions): DynamicModule {
    return {
      module: DatabaseModule,
      global: true,
      imports: options.imports || [],
      providers: [

          provide: 'DATABASE_OPTIONS',
          useFactory: options.useFactory,
          inject: options.inject || [],
        },
        DatabaseService,
      ],
      exports: [DatabaseService],
    };
  }
}

    // мӮ¬мҡ©н•ҳлҠ” мӘҪ
@Module({
  imports: [
    DatabaseModule.forRootAsync({
      imports: [ConfigModule],

    useFactoryм—җ м „лӢ¬лҗң н•ЁмҲҳлҠ” мқҳмЎҙм„ұ мЈјмһ… м»Ён…Ңмқҙл„Ҳк°Җ inject л°°м—ҙм—җ лӘ…мӢңлҗң н”„лЎңл°”мқҙлҚ”л“Өмқ„ мЈјмһ…н•ҙм„ң нҳём¶ңн•ңлӢӨ. ConfigModuleмқҙ importsм—җ мһҲм–ҙм•ј ConfigServiceлҘј мЈјмһ…л°ӣмқ„ мҲҳ мһҲлӢӨ.

    register / registerAsync¶

    forRootмҷҖ кө¬мЎ°лҠ” к°ҷм§Җл§Ң, мқҳлҜёк°Җ лӢӨлҘҙлӢӨ. forRootлҠ” м• н”ҢлҰ¬мјҖмқҙм…ҳ м „мІҙм—җм„ң н•ң лІҲ м„Өм •н•ҳлҠ” мҡ©лҸ„кі , registerлҠ” лӘЁл“Ҳлі„лЎң лӢӨлҘё м„Өм •мқ„ м Ғмҡ©н•  л•Ң мӮ¬мҡ©н•ңлӢӨ.

    @Module({})
export class HttpClientModule {
  static register(options: HttpClientOptions): DynamicModule {
    return {
      module: HttpClientModule,
      providers: [
        {
          provide: 'HTTP_CLIENT_OPTIONS',
          useValue: options,
        },
        HttpClientService,
      ],
      exports: [HttpClientService],
    };
  }
}

@Module({

    HttpClientModule.register({
      baseUrl: 'https://api.payment.com',

    м—¬кё°м„ң мЈјмқҳн•ҙм•ј н•  м җ: registerлЎң м—¬лҹ¬ лІҲ л“ұлЎқн•ҳл©ҙ к°Ғк°Ғ лі„лҸ„мқҳ лӘЁл“Ҳ мқёмҠӨн„ҙмҠӨк°Җ мғқкёҙлӢӨ. н•ҳм§Җл§Ң HttpClientServiceк°Җ к°ҷмқҖ нҶ нҒ°мқ„ мӮ¬мҡ©н•ҳкё° л•Ңл¬ём—җ, лӢӨлҘё лӘЁл“Ҳм—җм„ң HttpClientServiceлҘј мЈјмһ…л°ӣмңјл©ҙ м–ҙл–Ө мқёмҠӨн„ҙмҠӨк°Җ л“Өм–ҙмҳ¬м§Җ лӘЁнҳён•ҙм§„лӢӨ. мқҙ кІҪмҡ° к°Ғ лӘЁл“Ҳмқҙ мһҗмӢ мқҳ HttpClientServiceл§Ң мӮ¬мҡ©н•ҳлҸ„лЎқ exports лІ”мң„лҘј мһҳ кҙҖлҰ¬н•ҙм•ј н•ңлӢӨ.

    лҸҷм Ғ лӘЁл“Ҳ лӮҙл¶Җ л©”нғҖлҚ°мқҙн„° мЎ°н•© кіјм •¶

    NestJSк°Җ лҸҷм Ғ лӘЁл“Ҳмқ„ мІҳлҰ¬н•ҳлҠ” кіјм •мқ„ мЎ°кёҲ лҚ” кө¬мІҙм ҒмңјлЎң ліҙл©ҙ:

    1. imports л°°м—ҙм—җм„ң нҒҙлһҳмҠӨк°Җ м•„лӢҢ к°қмІҙ(DynamicModule)лҘј л§ҢлӮҳл©ҙ лҸҷм Ғ лӘЁл“ҲлЎң мІҳлҰ¬н•ңлӢӨ.
    2. DynamicModule к°қмІҙмқҳ module н•„л“ңлҘј лӘЁл“Ҳ нҶ нҒ°мңјлЎң мӮ¬мҡ©н•ңлӢӨ.
    3. н•ҙлӢ№ нҒҙлһҳмҠӨм—җ @Module() лҚ°мҪ”л Ҳмқҙн„°к°Җ мһҲмңјл©ҙ к·ё л©”нғҖлҚ°мқҙн„°мҷҖ DynamicModule к°қмІҙмқҳ л©”нғҖлҚ°мқҙн„°лҘј лі‘н•©н•ңлӢӨ.
    4. providers, controllers, imports, exportsк°Җ к°Ғк°Ғ н•©міҗм§„лӢӨ.
    // @Module лҚ°мҪ”л Ҳмқҙн„°мқҳ л©”нғҖлҚ°мқҙн„°
@Module({
  providers: [InternalHelper],
})
export class SomeModule {
  static forRoot(): DynamicModule {
    return {
      module: SomeModule,
      providers: [SomeService],  // мқҙкұҙ InternalHelperмҷҖ н•©міҗм§җ
      exports: [SomeService],
    };
  }
}

// кІ°кіјм ҒмңјлЎң SomeModuleмқҳ providers = [InternalHelper, SomeService]

    мқҙ лі‘н•© лҸҷмһ‘мқ„ лӘЁлҘҙл©ҙ "л¶„лӘ… forRootм—җм„ң providersлҘј м§Җм •н–ҲлҠ”лҚ° мҷң @Moduleмқҳ providersлҸ„ к°ҷмқҙ л“ұлЎқлҗҳм§Җ?"к°ҷмқҖ нҳјлһҖмқҙ мғқкёҙлӢӨ. л°ҳлҢҖлЎң, @Module({})мқ„ л№„мӣҢл‘җкі  лҸҷм Ғ л©”м„ңл“ңм—җм„ңл§Ң providersлҘј м§Җм •н•ҳл©ҙ к№”лҒ”н•ҳкІҢ кҙҖлҰ¬н•  мҲҳ мһҲлӢӨ.

    ModuleRef вҖ” лҹ°нғҖмһ„м—җ н”„лЎңл°”мқҙлҚ” кәјлӮҙкё°¶

    ModuleRefлҠ” DI м»Ён…Ңмқҙл„Ҳм—җм„ң лҹ°нғҖмһ„м—җ н”„лЎңл°”мқҙлҚ” мқёмҠӨн„ҙмҠӨлҘј м§Ғм ‘ мЎ°нҡҢн•ҳлҠ” к°қмІҙлӢӨ. мғқм„ұмһҗ мЈјмһ…л§ҢмңјлЎң н•ҙкІ°мқҙ м•Ҳ лҗҳлҠ” мғҒнҷ©м—җм„ң мӮ¬мҡ©н•ңлӢӨ.

    кё°ліё мӮ¬мҡ©лІ•¶

    import { Injectable, OnModuleInit } from '@nestjs/common';
import { ModuleRef } from '@nestjs/core';

@Injectable()
export class TaskRunner implements OnModuleInit {
  private reportService: ReportService;

  constructor(private readonly moduleRef: ModuleRef) {}

  onModuleInit() {
    // нҳ„мһ¬ лӘЁл“Ҳ м»Ён…ҚмҠӨнҠём—җм„ң н”„лЎңл°”мқҙлҚ”лҘј мЎ°нҡҢ
    this.reportService = this.moduleRef.get(ReportService, { strict: false });
  }
}

    get() л©”м„ңл“ңмқҳ л‘җ лІҲм§ё нҢҢлқјлҜён„° { strict: false }лҘј мЈјл©ҙ нҳ„мһ¬ лӘЁл“Ҳлҝҗ м•„лӢҲлқј м „м—ӯ м»Ён…Ңмқҙл„Ҳм—җм„ңлҸ„ кІҖмғүн•ңлӢӨ. кё°ліёк°’мқҖ strict: trueлЎң, нҳ„мһ¬ лӘЁл“Ҳ лӮҙм—җм„ңл§Ң м°ҫлҠ”лӢӨ. лӢӨлҘё лӘЁл“Ҳмқҳ н”„лЎңл°”мқҙлҚ”лҘј мЎ°нҡҢн•ҳл Өл©ҙ strict: falseк°Җ н•„мҡ”н•ңлҚ°, мқҙкұҙ н•ҙлӢ№ н”„лЎңл°”мқҙлҚ”к°Җ exportлҗң мғҒнғңм—¬м•ј лҸҷмһ‘н•ңлӢӨ.

    resolve() вҖ” REQUEST мҠӨмҪ”н”„ н”„лЎңл°”мқҙлҚ” мЎ°нҡҢ¶

    get()мқҖ мӢұкёҖнҶӨ н”„лЎңл°”мқҙлҚ”м—җл§Ң мӮ¬мҡ©н•ңлӢӨ. Scope.REQUESTлӮҳ Scope.TRANSIENTлЎң л“ұлЎқлҗң н”„лЎңл°”мқҙлҚ”лҠ” resolve()лҘј мҚЁм•ј н•ңлӢӨ.

    @Injectable()
export class AuditService {
  constructor(private readonly moduleRef: ModuleRef) {}

  async createAuditContext() {
    // resolve()лҠ” нҳём¶ңн•  л•Ңл§ҲлӢӨ мғҲ мқёмҠӨн„ҙмҠӨлҘј л§Ңл“ лӢӨ (transient)
    const logger = await this.moduleRef.resolve(TransientLogger);
    return logger;
  }

  async createSharedContext() {
    // contextIdлҘј м§Җм •н•ҳл©ҙ к°ҷмқҖ contextId лӮҙм—җм„ң мқёмҠӨн„ҙмҠӨлҘј кіөмң н•ңлӢӨ
    const contextId = ContextIdFactory.create();
    const a = await this.moduleRef.resolve(TransientLogger, contextId);
    const b = await this.moduleRef.resolve(TransientLogger, contextId);
    // a === b вҶ’ true (к°ҷмқҖ contextIdмқҙлҜҖлЎң)
  }
}

    resolve()лҠ” н•ӯмғҒ PromiseлҘј л°ҳнҷҳн•ңлӢӨ. get()кіј лӢӨлҘҙкІҢ лҸҷкё°к°Җ м•„лӢҲлӢҲк№Ң мғқм„ұмһҗм—җм„ң нҳём¶ңн•ҳл©ҙ м•Ҳ лҗҳкі , onModuleInitмқҙлӮҳ л№„мҰҲлӢҲмҠӨ лЎңм§Ғ м•Ҳм—җм„ң мҚЁм•ј н•ңлӢӨ.

    create() вҖ” DI м»Ён…Ңмқҙл„Ҳ л°–м—җм„ң мқёмҠӨн„ҙмҠӨ мғқм„ұ¶

    л“ұлЎқлҗҳм§Җ м•ҠмқҖ нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨлҘј DI м»Ён…Ңмқҙл„Ҳмқҳ мқҳмЎҙм„ұ мЈјмһ…мқ„ м Ғмҡ©н•ҙм„ң л§Ңл“Ө мҲҳ мһҲлӢӨ.

    мӢӨл¬ҙм—җм„ң ModuleRefк°Җ н•„мҡ”н•ң кІҪмҡ°¶

    лҢҖл¶Җл¶„мқҳ мғҒнҷ©м—җм„ңлҠ” мғқм„ұмһҗ мЈјмһ…мңјлЎң м¶©л¶„н•ҳлӢӨ. ModuleRefк°Җ н•„мҡ”н•ң мғҒнҷ©мқҖ м ңн•ңм ҒмқҙлӢӨ:

    • лҹ°нғҖмһ„м—җ нғҖмһ…мқҙ кІ°м •лҗҳлҠ” кІҪмҡ°: мқҙлІӨнҠё нғҖмһ…м—җ л”°лқј лӢӨлҘё н•ёл“Өлҹ¬лҘј м„ нғқн•ҙм•ј н•  л•Ң
    • мҲңнҷҳ мқҳмЎҙм„ұ нҡҢн”ј: forwardRef() лҢҖмӢ  лҹ°нғҖмһ„м—җ н•„мҡ”н•ң мӢңм җм—җ мЎ°нҡҢ
    • REQUEST мҠӨмҪ”н”„ н”„лЎңл°”мқҙлҚ”лҘј мӢұкёҖнҶӨм—җм„ң мӮ¬мҡ©н•ҙм•ј н•  л•Ң: мӢұкёҖнҶӨ м„ңл№„мҠӨк°Җ мҡ”мІӯлі„ м»Ён…ҚмҠӨнҠём—җ м ‘к·јн•ҙм•ј н•ҳлҠ” кІҪмҡ°
    // мқҙлІӨнҠё нғҖмһ…м—җ л”°лқј н•ёл“Өлҹ¬лҘј лҸҷм ҒмңјлЎң м„ нғқн•ҳлҠ” мҳҲмӢң
@Injectable()
export class EventDispatcher {
  private handlerMap: Map<string, Type<EventHandler>>;

  constructor(private readonly moduleRef: ModuleRef) {
    this.handlerMap = new Map([
      ['order.created', OrderCreatedHandler],
      ['order.cancelled', OrderCancelledHandler],
      ['payment.completed', PaymentCompletedHandler],
    ]);
  }

  async dispatch(event: DomainEvent) {
    const handlerType = this.handlerMap.get(event.type);
    if (!handlerType) {
      throw new Error(`н•ёл“Өлҹ¬ м—ҶмқҢ: ${event.type}`);
    }
    const handler = this.moduleRef.get(handlerType, { strict: false });
    await handler.handle(event);
  }
}

    LazyModuleLoader вҖ” лӘЁл“Ҳ м§Җм—° лЎңл”©¶

    LazyModuleLoaderлҠ” л¶ҖнҠёмҠӨнҠёлһ© мӢңм җм—җ лӘЁл“Ҳмқ„ лЎңл”©н•ҳм§Җ м•Ҡкі , мӢӨм ңлЎң н•„мҡ”н•ҙм§ҖлҠ” мӢңм җм—җ лЎңл”©н•ҳлҠ” кё°лҠҘмқҙлӢӨ. мҪңл“ң мҠӨнғҖнҠё мӢңк°„мқҙ мӨ‘мҡ”н•ң м„ңлІ„лҰ¬мҠӨ нҷҳкІҪмқҙлӮҳ, нҠ№м • мЎ°кұҙм—җм„ңл§Ң мӮ¬мҡ©н•ҳлҠ” л¬ҙкұ°мҡҙ лӘЁл“Ҳмқҙ мһҲмқ„ л•Ң мң мҡ©н•ҳлӢӨ.

    кё°ліё мӮ¬мҡ©лІ•¶

    import { Injectable } from '@nestjs/common';

@Injectable()
export class ReportController {

    const reportService = moduleRef.get(ReportService);

    load()лҠ” нҢ©нҶ лҰ¬ н•ЁмҲҳлҘј л°ӣлҠ”лӢӨ. нҷ”мӮҙн‘ң н•ЁмҲҳлЎң к°җмӢём•ј н•ҳл©°, м§Ғм ‘ нҒҙлһҳмҠӨлҘј л„ҳкё°л©ҙ м•Ҳ лҗңлӢӨ. н•ң лІҲ лЎңл”©лҗң лӘЁл“ҲмқҖ мәҗмӢңлҗҳм–ҙ л‘җ лІҲм§ё load() нҳём¶ңл¶Җн„°лҠ” мҰүмӢң л°ҳнҷҳлҗңлӢӨ.

    м§Җм—° лЎңл”© мӢң мЈјмқҳмӮ¬н•ӯ¶

    м§Җм—° лЎңл”©лҗң лӘЁл“ҲмқҖ лӘҮ к°Җм§Җ м ңм•Ҫмқҙ мһҲлӢӨ:

    м»ЁнҠёлЎӨлҹ¬к°Җ л“ұлЎқлҗҳм§Җ м•ҠлҠ”лӢӨ. м§Җм—° лЎңл”©лҗң лӘЁл“Ҳм—җ м„ м–ёлҗң м»ЁнҠёлЎӨлҹ¬лҠ” лқјмҡ°нҠёлЎң л“ұлЎқлҗҳм§Җ м•ҠлҠ”лӢӨ. HTTP лқјмҡ°нҢ…мқҖ л¶ҖнҠёмҠӨнҠёлһ© мӢңм җм—җ кІ°м •лҗҳкё° л•Ңл¬ёмқҙлӢӨ. м§Җм—° лЎңл”©мқҖ н”„лЎңл°”мқҙлҚ” мң„мЈјмқҳ л°ұк·ёлқјмҡҙл“ң мһ‘м—…м—җ м Ғн•©н•ҳлӢӨ.

    global: true лӘЁл“ҲмқҖ м§Җм—° лЎңл”©н•  мҲҳ м—ҶлӢӨ. кёҖлЎңлІҢ лӘЁл“ҲмқҖ л¶ҖнҠёмҠӨнҠёлһ© мӢңм җм—җ мқҙлҜё м „м—ӯ м»Ён…Ңмқҙл„Ҳм—җ л“ұлЎқлҗҳм–ҙм•ј н•ҳкё° л•Ңл¬ёмқҙлӢӨ.

    лқјмқҙн”„мӮ¬мқҙнҒҙ нӣ…мқҖ м •мғҒ лҸҷмһ‘н•ңлӢӨ. OnModuleInitкіј OnApplicationBootstrapмқҙ load() мӢңм җм—җ мӢӨн–үлҗңлӢӨ. лӢЁ, OnModuleDestroyлҠ” м• н”ҢлҰ¬мјҖмқҙм…ҳ мў…лЈҢ мӢң нҳём¶ңлҗңлӢӨ.

    м§Ғм ‘ кө¬нҳ„н•ҳлҠ” кІҢ лӮҳмқҖ кІҪмҡ°лҸ„ мһҲлӢӨ.   private validators: Validator[] = [];

  constructor(private readonly discoveryService: DiscoveryService) {}

  onModuleInit() {
    const providers = this.discoveryService.getProviders();
    this.validators = providers
      .filter(wrapper => {
        if (!wrapper.metatype) return false;
        return Reflect.getMetadata(VALIDATOR_KEY, wrapper.metatype);
      })
      .map(wrapper => wrapper.instance as Validator);
  }

  async validateAll(input: any) {
    for (const validator of this.validators) {
      validator.validate(input);
    }
  }
}

    мӢӨл¬ҙ Graceful Shutdown кө¬нҳ„ мҳҲмӢң¶

    @Injectable()
export class GracefulShutdownService
  implements OnModuleDestroy, BeforeApplicationShutdown
{
  private isShuttingDown = false;
  private activeRequests = 0;

  onRequest() {
    this.activeRequests++;
  }

  onRequestComplete() {
    this.activeRequests--;
  }

  getIsShuttingDown() {
    return this.isShuttingDown;
  }

  async onModuleDestroy() {
    this.isShuttingDown = true;
    // мқҙ мӢңм җл¶Җн„° мғҲ мҡ”мІӯмқ„ л°ӣм§Җ м•ҠлҸ„лЎқ н”Ңлһҳк·ё м„Өм •
    // лЎңл“ңл°ёлҹ°м„ң н—¬мҠӨмІҙнҒ¬к°Җ мқҙ к°’мқ„ ліҙкі  нҠёлһҳн”Ҫмқ„ л№јмӨҢ
  }

  async beforeApplicationShutdown(signal?: string) {
    console.log(`${signal} мҲҳмӢ , м§„н–ү мӨ‘мқё мҡ”мІӯ ${this.activeRequests}к°ң лҢҖкё°`);

    // м§„н–ү мӨ‘мқё мҡ”мІӯмқҙ лҒқлӮ  л•Ңк№Ңм§Җ лҢҖкё°
    const maxWait = 30_000;  // мөңлҢҖ 30мҙҲ
    const start = Date.now();

    while (this.activeRequests > 0 && Date.now() - start < maxWait) {
      await new Promise(resolve => setTimeout(resolve, 100));
    }

    if (this.activeRequests > 0) {
      console.warn(`нғҖмһ„м•„мӣғ: ${this.activeRequests}к°ң мҡ”мІӯмқҙ к°•м ң мў…лЈҢлҗЁ`);
    }
  }
}

    2. OnModuleDestroyм—җм„ң DBлҘј лЁјм Җ лҒҠм–ҙлІ„лҰ¬лҠ” кІҪмҡ°

    OnModuleDestroy мӢӨн–ү мҲңм„ң:
  1. DatabaseModule.onModuleDestroy() вҶ’ м»Өл„Ҙм…ҳ н’Җ мў…лЈҢ
  2. LogModule.onModuleDestroy() вҶ’ лІ„нҚјм—җ мҢ“мқё лЎңк·ёлҘј DBм—җ flushн•ҳл ӨлҠ”лҚ° мӢӨнҢЁ

    imports мҲңм„ңк°Җ [DatabaseModule, LogModule]мқҙл©ҙ destroyлҠ” м—ӯмҲңмңјлЎң LogModule вҶ’ DatabaseModuleмқҙ лҗҳм–ҙм•ј н•  кІғ к°ҷм§Җл§Ң, NestJSмқҳ мӢӨм ң лҸҷмһ‘м—җм„ң мқҙ мҲңм„ңк°Җ н•ӯмғҒ ліҙмһҘлҗҳм§Җ м•Ҡмқ„ мҲҳ мһҲлӢӨ. м•Ҳм „н•ҳкІҢ мІҳлҰ¬н•ҳл Өл©ҙ BeforeApplicationShutdownм—җм„ң лЎңк·ё flushлҘј н•ҳкі , OnModuleDestroyм—җм„ң DBлҘј лҒҠлҠ” мӢқмңјлЎң нӣ…мқ„ л¶„лҰ¬н•ҙм•ј н•ңлӢӨ.

    3. мҝ лІ„л„ӨнӢ°мҠӨм—җм„ң SIGTERM нғҖмқҙл°Қ л¬ём ң

    мҝ лІ„л„ӨнӢ°мҠӨлҠ” Podмқ„ мў…лЈҢн•  л•Ң SIGTERMмқ„ ліҙлӮҙкі  terminationGracePeriodSeconds(кё°ліё 30мҙҲ) нӣ„м—җ SIGKILLмқ„ ліҙлӮёлӢӨ. NestJSмқҳ Graceful Shutdownмқҙ 30мҙҲ м•Ҳм—җ лҒқлӮҳм§Җ м•Ҡмңјл©ҙ SIGKILLлЎң к°•м ң мў…лЈҢлҗңлӢӨ.

    - name: app

        preStop:
          exec:
