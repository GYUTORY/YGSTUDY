    Service¶

    // products/products.service.ts
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Product } from './entities/product.entity';
import { CreateProductDto } from './dto/create-product.dto';
import { UpdateProductDto } from './dto/update-product.dto';

@Injectable()
export class ProductsService {
  constructor(
    @InjectRepository(Product)
    private readonly productRepo: Repository<Product>,
  ) {}

  async create(dto: CreateProductDto): Promise<Product> {
    const product = this.productRepo.create(dto);
    return this.productRepo.save(product);
  }

  async findAll(): Promise<Product[]> {
    return this.productRepo.find();
  }

  async findOne(id: number): Promise<Product> {
    const product = await this.productRepo.findOne({ where: { id } });
    if (!product) {
      throw new NotFoundException(`Product #${id} not found`);
    }
    return product;
  }

  async update(id: number, dto: UpdateProductDto): Promise<Product> {
    const product = await this.findOne(id);
    Object.assign(product, dto);
    return this.productRepo.save(product);
  }

  async remove(id: number): Promise<void> {
    const result = await this.productRepo.delete(id);
    if (result.affected === 0) {
      throw new NotFoundException(`Product #${id} not found`);
    }
  }
}

    update 메서드에서 productRepo.update()가 아닌 findOne + save를 쓴 이유: update()는 Entity 훅(@BeforeUpdate 등)을 실행하지 않는다. 훅이 필요 없는 경우에만 update()를 직접 쓴다.

  Post,

import { UpdateProductDto } from './dto/update-product.dto';

@Controller('products')
export class ProductsController {

  @Post()
  @HttpCode(HttpStatus.CREATED)
  create(@Body() dto: CreateProductDto) {
    return this.productsService.create(dto);
  }

  @Get()
  findAll() {
    return this.productsService.findAll();
  }

  @Get(':id')
  findOne(@Param('id', ParseIntPipe) id: number) {
    return this.productsService.findOne(id);
  }

  @Patch(':id')
  update(
    @Param('id', ParseIntPipe) id: number,
    @Body() dto: UpdateProductDto,
  ) {
    return this.productsService.update(id, dto);
  }

  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)

    N+1 쿼리 문제¶

    NestJS + TypeORM 조합에서 가장 흔하게 겪는 성능 문제다.

    문제 상황¶

    // 유저 목록을 가져오면서 각 유저의 게시글도 필요한 경우
async findAll(): Promise<User[]> {
  const users = await this.userRepo.find(); // SELECT * FROM users

  // 이 시점에서 user.posts에 접근하면?
  // TypeORM은 lazy loading이 기본이 아니라 undefined가 나온다.
  // eager: true로 설정하면 유저마다 SELECT가 한 번씩 나간다 → N+1
  return users;
}

    Entity에 eager: true를 설정하면 find() 호출 시 자동으로 join이 걸리지만, 모든 조회에서 join이 발생한다. 게시글이 필요 없는 조회에서도 join이 걸린다.

    해결: 필요한 곳에서만 relations 지정¶

    // posts가 필요한 조회
async findAllWithPosts(): Promise<User[]> {
  return this.userRepo.find({
    relations: ['posts'],
  });
  // SELECT ... FROM users LEFT JOIN posts ON ...
  // 쿼리 한 방에 끝난다
}

// posts가 필요 없는 조회
async findAll(): Promise<User[]> {
  return this.userRepo.find();
  // SELECT * FROM users
}

    QueryBuilder로 세밀한 제어¶

    async findUsersWithRecentPosts(): Promise<User[]> {
  return this.userRepo
    .createQueryBuilder('user')
    .leftJoinAndSelect('user.posts', 'post', 'post.createdAt > :date', {
      date: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000),
    })
    .getMany();
}

    find({ relations }) 방식은 단순 join만 되지만, QueryBuilder는 join 조건을 걸 수 있다. 최근 7일 게시글만 가져오는 식의 조건부 로딩이 필요하면 QueryBuilder를 쓴다.

    주의할 점¶

    relations에 중첩 관계를 넣으면 쿼리가 급격히 무거워진다.

    // 이렇게 쓰면 3단계 join이 발생
this.userRepo.find({
  relations: ['posts', 'posts.comments', 'posts.comments.author'],
});

    중첩이 깊어지면 응답 JSON 크기도 커진다. 필요한 데이터만 select로 골라서 가져오는 게 낫다.

    트랜잭션 처리¶

    여러 테이블에 데이터를 동시에 넣어야 할 때, 중간에 하나라도 실패하면 전부 롤백해야 한다.

    QueryRunner 방식¶

    import { Injectable } from '@nestjs/common';
import { DataSource } from 'typeorm';
import { Order } from './entities/order.entity';
import { OrderItem } from './entities/order-item.entity';
import { Product } from '../products/entities/product.entity';

@Injectable()
export class OrdersService {
  constructor(private readonly dataSource: DataSource) {}

  async createOrder(userId: number, items: { productId: number; quantity: number }[]) {
    const queryRunner = this.dataSource.createQueryRunner();
    await queryRunner.connect();
    await queryRunner.startTransaction();

    try {
      // 1. 주문 생성
      const order = queryRunner.manager.create(Order, { userId });
      const savedOrder = await queryRunner.manager.save(order);

      // 2. 주문 항목 생성 + 재고 차감
      for (const item of items) {
        const product = await queryRunner.manager.findOne(Product, {
          where: { id: item.productId },
          lock: { mode: 'pessimistic_write' }, // 동시 주문 시 재고 꼬임 방지
        });

        if (!product || product.stock < item.quantity) {
          throw new Error(`재고 부족: ${item.productId}`);
        }

        product.stock -= item.quantity;
        await queryRunner.manager.save(product);

        const orderItem = queryRunner.manager.create(OrderItem, {
          orderId: savedOrder.id,
          productId: item.productId,
          quantity: item.quantity,
          price: product.price,
        });
        await queryRunner.manager.save(orderItem);
      }

      await queryRunner.commitTransaction();
      return savedOrder;
    } catch (err) {
      await queryRunner.rollbackTransaction();
      throw err;
    } finally {
      await queryRunner.release(); // 반드시 release 해야 한다. 안 하면 커넥션이 풀에 반환되지 않는다
    }
  }
}

    pessimistic_write 락은 해당 row를 다른 트랜잭션이 읽지 못하게 잠근다. 동시에 같은 상품에 주문이 들어왔을 때 재고가 마이너스로 빠지는 걸 방지한다.

    transaction() 메서드 방식¶

    QueryRunner를 직접 다루는 게 번거로우면 DataSource.transaction()을 쓸 수 있다. rollback/release를 자동으로 처리해준다.

    async createOrder(userId: number, items: { productId: number; quantity: number }[]) {
  return this.dataSource.transaction(async (manager) => {
    const order = manager.create(Order, { userId });
    const savedOrder = await manager.save(order);

    for (const item of items) {
      const product = await manager.findOne(Product, {
        where: { id: item.productId },
        lock: { mode: 'pessimistic_write' },
      });

      if (!product || product.stock < item.quantity) {
        throw new Error(`재고 부족: ${item.productId}`);
      }

      product.stock -= item.quantity;
      await manager.save(product);

      await manager.save(
        manager.create(OrderItem, {
          orderId: savedOrder.id,
          productId: item.productId,
          quantity: item.quantity,
          price: product.price,
        }),
      );
    }

    return savedOrder;
  });
}

    콜백 안에서 throw하면 자동으로 rollback된다. 대부분의 경우 이 방식이 더 깔끔하다.

    트랜잭션에서 흔한 실수¶

    트랜잭션 밖의 repository를 쓰는 경우:

    // 잘못된 예
async createOrder(userId: number) {
  return this.dataSource.transaction(async (manager) => {
    const order = manager.create(Order, { userId });
    await manager.save(order);

    // this.productRepo는 트랜잭션 밖의 커넥션을 쓴다!
    // 트랜잭션 격리가 깨진다
    await this.productRepo.update(1, { stock: 10 });
  });
}

    트랜잭션 안에서는 반드시 manager를 통해 쿼리해야 한다. DI로 주입받은 repository를 쓰면 별도의 커넥션으로 쿼리가 나가서 트랜잭션이 의미 없어진다.

    Auth Service¶

    // auth/auth.service.ts
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { UsersService } from '../users/users.service';
import * as bcrypt from 'bcrypt';

@Injectable()
export class AuthService {
  constructor(
    private readonly usersService: UsersService,
    private readonly jwtService: JwtService,
  ) {}

  async validateUser(email: string, password: string) {
    const user = await this.usersService.findByEmail(email);
    if (!user) {
      throw new UnauthorizedException('이메일 또는 비밀번호가 틀렸다');
    }

    const isMatch = await bcrypt.compare(password, user.password);
    if (!isMatch) {
      throw new UnauthorizedException('이메일 또는 비밀번호가 틀렸다');
    }

    const { password: _, ...result } = user;
    return result;
  }

  async login(email: string, password: string) {
    const user = await this.validateUser(email, password);
    const payload = { sub: user.id, email: user.email };
    return {
      accessToken: this.jwtService.sign(payload),
    };
  }

  async register(email: string, password: string, name: string) {
    const hashedPassword = await bcrypt.hash(password, 10);
    const user = await this.usersService.create({
      email,
      password: hashedPassword,
      name,
    });
    const { password: _, ...result } = user;
    return result;
  }
}

    validateUser에서 "이메일이 없다" / "비밀번호가 틀렸다"를 구분해서 에러를 보내면 공격자에게 해당 이메일이 가입되어 있다는 정보를 준다. 둘 다 같은 메시지를 쓴다.

    JWT Strategy¶

    // auth/strategies/jwt.strategy.ts
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';
import { UsersService } from '../../users/users.service';

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(
    private readonly usersService: UsersService,
    configService: ConfigService,
  ) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      ignoreExpiration: false,
      secretOrKey: configService.get('JWT_SECRET'),
    });
  }

  async validate(payload: { sub: number; email: string }) {
    const user = await this.usersService.findOne(payload.sub);
    if (!user) {
      throw new UnauthorizedException();
    }
    return { userId: user.id, email: user.email };
  }
}

    validate 메서드가 리턴하는 객체가 req.user에 담긴다. 여기에 민감한 정보(비밀번호 해시 등)를 넣으면 안 된다.

    E2E Test와 DB 초기화¶

    E2E 테스트에서 가장 골치 아픈 부분이 DB 상태 관리다. 테스트마다 깨끗한 DB에서 시작해야 결과가 예측 가능하다.

    // test/helpers/db-cleanup.ts
import { DataSource } from 'typeorm';

export async function cleanupDatabase(dataSource: DataSource) {
  const entities = dataSource.entityMetadatas;

  for (const entity of entities) {
    const repository = dataSource.getRepository(entity.name);
    await repository.query(`TRUNCATE TABLE "${entity.tableName}" CASCADE`);
  }
}

    // test/users.e2e-spec.ts
import { Test, TestingModule } from '@nestjs/testing';
import { INestApplication, ValidationPipe } from '@nestjs/common';
import * as request from 'supertest';
import { AppModule } from '../src/app.module';
import { DataSource } from 'typeorm';
import { cleanupDatabase } from './helpers/db-cleanup';

describe('UsersController (e2e)', () => {
  let app: INestApplication;
  let dataSource: DataSource;

  beforeAll(async () => {
    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();

    app = moduleFixture.createNestApplication();
    app.useGlobalPipes(new ValidationPipe({ whitelist: true }));
    await app.init();

    dataSource = app.get(DataSource);
  });

  beforeEach(async () => {
    await cleanupDatabase(dataSource);
  });

    await app.close();

  describe('GET /users', () => {

    주의사항:

    • TRUNCATE CASCADE는 FK 관계가 있는 테이블까지 같이 비운다. MySQL에서는 SET FOREIGN_KEY_CHECKS = 0 후 truncate하고 다시 SET FOREIGN_KEY_CHECKS = 1로 켜야 한다.
    • beforeAll에서 앱을 초기화하고 beforeEach에서 DB를 비우는 패턴이 표준이다. beforeEach에서 앱을 매번 초기화하면 테스트가 느려진다.
    • E2E 테스트는 테스트 전용 DB를 쓴다. 개발 DB에 돌리면 데이터가 날아간다. .env.test 파일을 분리하든, 환경변수로 테스트 DB를 지정하든 방법을 정해둔다.

    E2E 테스트에서 ValidationPipe¶

    main.ts에서 app.useGlobalPipes()로 등록한 ValidationPipe는 E2E 테스트의 app에는 적용되지 않는다. 테스트에서도 별도로 등록해야 한다. 빠뜨리면 잘못된 DTO가 그대로 통과해서 테스트가 의미 없어진다.

app.useGlobalPipes(
  new ValidationPipe({
    whitelist: true,    // DTO에 없는 필드 자동 제거
    forbidNonWhitelisted: true, // DTO에 없는 필드가 오면 400
    transform: true,    // 쿼리 파라미터 자동 타입 변환
  }),
);

import { randomUUID } from 'crypto';

@Controller('files')
export class FilesController {
  @Post('upload')
  @UseInterceptors(
    FileInterceptor('file', {
      storage: diskStorage({
        destination: './uploads',
        filename: (req, file, cb) => {
          const name = `${randomUUID()}${extname(file.originalname)}`;
          cb(null, name);
        },
      }),
    }),
  )
  upload(
    @UploadedFile(
      new ParseFilePipe({
        validators: [
          new MaxFileSizeValidator({ maxSize: 5 * 1024 * 1024 }), // 5MB
          new FileTypeValidator({ fileType: /image\/(jpg|jpeg|png|gif)/ }),
        ],
      }),
    )
    file: Express.Multer.File,
  ) {
    return {
      filename: file.filename,
      size: file.size,
    };
  }
}

    Gateway¶

    // chat/chat.gateway.ts
import {
  WebSocketGateway,
  WebSocketServer,
  SubscribeMessage,
  OnGatewayConnection,
  OnGatewayDisconnect,
  ConnectedSocket,
  MessageBody,
} from '@nestjs/websockets';
import { Server, Socket } from 'socket.io';

@WebSocketGateway({ cors: { origin: '*' } })
export class ChatGateway implements OnGatewayConnection, OnGatewayDisconnect {
  @WebSocketServer()
  server: Server;

  private connectedClients = new Map<string, string>(); // socketId → userId

  handleConnection(client: Socket) {
    console.log(`connected: ${client.id}`);
  }

  handleDisconnect(client: Socket) {
    this.connectedClients.delete(client.id);
    console.log(`disconnected: ${client.id}`);
  }

  @SubscribeMessage('joinRoom')
  handleJoin(
    @ConnectedSocket() client: Socket,
    @MessageBody() room: string,
  ) {
    client.join(room);
    client.to(room).emit('userJoined', { socketId: client.id });
  }

  @SubscribeMessage('message')
  handleMessage(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { room: string; message: string },
  ) {
    // 본인을 제외한 방의 다른 사람에게 전송
    client.to(data.room).emit('message', {
      from: client.id,
      message: data.message,
    });
  }
}

    @WebSocketGateway()에 CORS 설정을 안 하면 브라우저에서 연결이 안 된다. 프로덕션에서는 origin: '*' 대신 실제 도메인을 지정한다.

    this.server.emit()은 모든 클라이언트에게, client.to(room).emit()은 방의 다른 클라이언트에게, client.emit()은 본인에게만 전송한다. 셋을 헷갈리면 메시지가 이상한 곳으로 간다.
