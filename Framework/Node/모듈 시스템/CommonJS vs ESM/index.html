    ECMAScript Modules (ESM) мғҒм„ё л¶„м„қ¶

    ESMмқҳ н•өмӢ¬ к°ңл…җ¶

    ESMмқҖ JavaScript н‘ңмӨҖ лӘЁл“Ҳ мӢңмҠӨн…ңмңјлЎң, ES6(ES2015)м—җм„ң лҸ„мһ…лҗҳм—ҲмҠөлӢҲлӢӨ. ESMмқҳ н•өмӢ¬мқҖ м •м Ғ лӘЁл“Ҳ кө¬мЎ°мҷҖ м»ҙнҢҢмқј нғҖмһ„ мқҳмЎҙм„ұ л¶„м„қмһ…лӢҲлӢӨ.

    ESMмқҳ мһ‘лҸҷ мӣҗлҰ¬¶

    1. м •м Ғ л¶„м„қ: лӘЁл“Ҳмқҳ мқҳмЎҙм„ұмқҙ мҪ”л“ң мӢӨн–ү м „м—җ кІ°м •лҗЁ
    2. л№„лҸҷкё° лЎңл”©: лӘЁл“Ҳмқ„ л№„лҸҷкё°м ҒмңјлЎң лЎңл“ң к°ҖлҠҘ
    3. нҠёлҰ¬ мүҗмқҙнӮ№: мӮ¬мҡ©н•ҳм§Җ м•ҠлҠ” мҪ”л“ң м ңкұ° к°ҖлҠҘ
    4. мҲңнҷҳ мқҳмЎҙм„ұ н•ҙкІ°: м •м Ғ л¶„м„қмңјлЎң мҲңнҷҳ мқҳмЎҙм„ұ л¬ём ң н•ҙкІ°
    // ESMмқҳ м •м Ғ кө¬мЎ° мҳҲмӢң

    мһҘм җ: - н‘ңмӨҖнҷ”: JavaScript кіөмӢқ н‘ңмӨҖ - лёҢлқјмҡ°м Җ м§Җмӣҗ: л„ӨмқҙнӢ°лёҢ лёҢлқјмҡ°м Җ м§Җмӣҗ - нҠёлҰ¬ мүҗмқҙнӮ№: лІҲл“Ө нҒ¬кё° мөңм Ғнҷ” - м •м Ғ л¶„м„қ: IDE м§Җмӣҗ л°Ҹ лҸ„кө¬ нҷңмҡ© к°ҖлҠҘ - л№„лҸҷкё° лЎңл”©: м„ұлҠҘ мөңм Ғнҷ” к°ҖлҠҘ

    лӢЁм җ: - ліөмһЎм„ұ: мғҒлҢҖм ҒмңјлЎң ліөмһЎн•ң л¬ёлІ• - нҳёнҷҳм„ұ: мқјл¶Җ кө¬нҳ• нҷҳкІҪм—җм„ң м§Җмӣҗ м ңн•ң - м„Өм • н•„мҡ”: Node.jsм—җм„ң м¶”к°Җ м„Өм • н•„мҡ” - мҲңнҷҳ мқҳмЎҙм„ұ: ліөмһЎн•ң мҲңнҷҳ мқҳмЎҙм„ұ мІҳлҰ¬

const Module = require('module');
const originalRequire = Module.prototype.require;
const loadingModules = new Set();

Module.prototype.require = function(id) {
    if (loadingModules.has(id)) {
        console.warn(`мҲңнҷҳ мқҳмЎҙм„ұ к°җм§Җ: ${id}`);
    }
    loadingModules.add(id);
    const result = originalRequire.call(this, id);
    loadingModules.delete(id);
    return result;
};
