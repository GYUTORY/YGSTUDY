# 비트 단위 논리 연산

## 들어가며

컴퓨터의 모든 데이터는 0과 1의 조합이다. 비트 단위 논리 연산은 이 비트들을 직접 조작하는 연산이다.

고수준 언어만 쓰면 비트 연산을 만질 일이 별로 없지만, 권한 시스템, 네트워크 서브넷 계산, 피처 플래그 같은 걸 구현하다 보면 반드시 만나게 된다. 이론보다 실제 코드 위주로 정리한다.

## 기본 연산자 요약

| 연산자 | 이름 | 동작 | 예시 (4비트) |
|--------|------|------|-------------|
| `&` | AND | 둘 다 1이면 1 | `1010 & 1100 = 1000` |
| `\|` | OR | 하나라도 1이면 1 | `1010 \| 1100 = 1110` |
| `^` | XOR | 서로 다르면 1 | `1010 ^ 1100 = 0110` |
| `~` | NOT | 비트 반전 | `~1010 = 0101` |
| `<<` | 왼쪽 시프트 | 왼쪽으로 밀기 (빈자리 0) | `0011 << 1 = 0110` |
| `>>` | 오른쪽 시프트 | 오른쪽으로 밀기 | `1100 >> 1 = 0110` |

## AND 연산 (&) — 비트 추출과 마스킹

AND는 두 비트가 모두 1일 때만 1이 된다. 핵심 용도는 **특정 비트만 꺼내기(마스킹)**와 **특정 비트를 0으로 끄기**다.

```java
// 하위 4비트만 추출
int value = 0b11010110; // 214
int lower4 = value & 0x0F; // 0b00000110 = 6
```

```python
# 짝수/홀수 판별 — 최하위 비트가 0이면 짝수
def is_even(n):
    return (n & 1) == 0

is_even(42)  # True
is_even(7)   # False
```

## OR 연산 (|) — 비트 설정

OR은 두 비트 중 하나라도 1이면 1이 된다. 핵심 용도는 **특정 비트를 1로 켜기**다.

```java
// 3번째 비트를 켜기
int flags = 0b0000;
flags = flags | (1 << 2); // 0b0100
```

## XOR 연산 (^) — 토글과 비교

XOR은 두 비트가 서로 다르면 1이 된다. 같은 값으로 두 번 XOR하면 원래 값으로 돌아오는 **가역성**이 핵심이다.

```javascript
// 특정 비트 토글 (켜져 있으면 끄고, 꺼져 있으면 켜기)
let flags = 0b1010;
flags = flags ^ 0b0010; // 0b1000 — 1번 비트 꺼짐
flags = flags ^ 0b0010; // 0b1010 — 다시 켜짐
```

## NOT 연산 (~) — 비트 반전

모든 비트를 뒤집는다. 마스크를 만들 때 자주 쓴다.

```java
// 특정 비트만 0으로 끄는 마스크 생성
int FLAG = 1 << 3; // 0b1000
int mask = ~FLAG;   // 0b...11110111

int flags = 0b1111;
flags = flags & mask; // 0b0111 — 3번 비트만 꺼짐
```

## 시프트 연산

왼쪽 시프트(`<<`)는 2의 거듭제곱 곱셈, 오른쪽 시프트(`>>`)는 2의 거듭제곱 나눗셈과 같다.

```python
# 시프트로 곱셈/나눗셈
x = 5
x << 1  # 10 (5 * 2)
x << 3  # 40 (5 * 8)
x >> 1  # 2  (5 / 2, 소수점 버림)
```

Java에는 부호 없는 오른쪽 시프트(`>>>`)가 있다. 부호 비트와 관계없이 왼쪽에 항상 0을 채운다. Python은 정수 크기가 가변이라 `>>>` 연산자가 없다.

```java
int negative = -1;         // 0xFFFFFFFF
negative >> 1;             // 0xFFFFFFFF (-1) — 부호 비트 유지
negative >>> 1;            // 0x7FFFFFFF (2147483647) — 0으로 채움
```

## 실무 예제 1: 플래그 기반 권한 시스템

파일 시스템이나 API에서 권한을 관리할 때 비트 플래그를 쓰면 하나의 정수로 여러 권한을 표현할 수 있다. 리눅스 파일 퍼미션(`chmod 755`)이 대표적이다.

### Java

```java
public class Permission {
    static final int READ    = 1 << 0; // 0b0001 = 1
    static final int WRITE   = 1 << 1; // 0b0010 = 2
    static final int EXECUTE = 1 << 2; // 0b0100 = 4
    static final int DELETE  = 1 << 3; // 0b1000 = 8

    private int permissions;

    public Permission(int permissions) {
        this.permissions = permissions;
    }

    // 권한 부여 — OR로 비트를 켠다
    public void grant(int perm) {
        permissions |= perm;
    }

    // 권한 회수 — AND + NOT으로 비트를 끈다
    public void revoke(int perm) {
        permissions &= ~perm;
    }

    // 권한 확인 — AND로 해당 비트가 켜져 있는지 본다
    public boolean has(int perm) {
        return (permissions & perm) == perm;
    }

    // 권한 토글 — XOR로 반전
    public void toggle(int perm) {
        permissions ^= perm;
    }

    public static void main(String[] args) {
        Permission user = new Permission(READ | WRITE);

        System.out.println(user.has(READ));     // true
        System.out.println(user.has(EXECUTE));  // false

        user.grant(EXECUTE);
        System.out.println(user.has(EXECUTE));  // true

        user.revoke(WRITE);
        System.out.println(user.has(WRITE));    // false

        // 여러 권한을 한번에 확인
        System.out.println(user.has(READ | EXECUTE)); // true
    }
}
```

### JavaScript

```javascript
const ROLE = {
  VIEW:    1 << 0,  // 1
  EDIT:    1 << 1,  // 2
  DELETE:  1 << 2,  // 4
  ADMIN:   1 << 3,  // 8
};

// 역할 조합
const EDITOR = ROLE.VIEW | ROLE.EDIT;           // 3
const MODERATOR = ROLE.VIEW | ROLE.EDIT | ROLE.DELETE; // 7
const SUPER_ADMIN = ROLE.VIEW | ROLE.EDIT | ROLE.DELETE | ROLE.ADMIN; // 15

function hasPermission(userRole, requiredPerm) {
  return (userRole & requiredPerm) === requiredPerm;
}

let currentUser = EDITOR;

hasPermission(currentUser, ROLE.VIEW);    // true
hasPermission(currentUser, ROLE.DELETE);  // false

// 권한 추가
currentUser |= ROLE.DELETE;
hasPermission(currentUser, ROLE.DELETE);  // true

// 권한 제거
currentUser &= ~ROLE.EDIT;
hasPermission(currentUser, ROLE.EDIT);    // false
```

### Python

```python
from enum import IntFlag

class Perm(IntFlag):
    READ    = 1 << 0
    WRITE   = 1 << 1
    EXECUTE = 1 << 2
    DELETE  = 1 << 3

# IntFlag를 쓰면 비트 연산이 그대로 동작한다
user_perm = Perm.READ | Perm.WRITE

Perm.READ in user_perm      # True
Perm.EXECUTE in user_perm   # False

# 권한 추가/제거
user_perm |= Perm.EXECUTE
user_perm &= ~Perm.WRITE

print(user_perm)  # Perm.READ|EXECUTE
```

**주의할 점**: 플래그 수가 32개(int 기준)를 넘어가면 `long`이나 `BigInteger`를 써야 한다. Java에서는 `EnumSet`이 내부적으로 비트 플래그를 쓰면서 타입 안전성도 제공하니, 실무에서는 `EnumSet`을 먼저 검토하는 게 낫다.

## 실무 예제 2: IP 서브넷 계산

네트워크에서 IP 주소와 서브넷 마스크의 AND 연산은 네트워크 주소를 구하는 기본 연산이다.

### Java

```java
public class SubnetCalculator {

    // "192.168.1.100" -> int 변환
    static int ipToInt(String ip) {
        String[] parts = ip.split("\\.");
        int result = 0;
        for (String part : parts) {
            result = (result << 8) | Integer.parseInt(part);
        }
        return result;
    }

    // int -> "192.168.1.100" 변환
    static String intToIp(int ip) {
        return ((ip >> 24) & 0xFF) + "." +
               ((ip >> 16) & 0xFF) + "." +
               ((ip >> 8) & 0xFF)  + "." +
               (ip & 0xFF);
    }

    // CIDR 표기에서 서브넷 마스크 생성
    // /24 -> 0xFFFFFF00 (255.255.255.0)
    static int cidrToMask(int prefix) {
        // -1은 0xFFFFFFFF(모든 비트 1)
        // 왼쪽 시프트로 하위 비트를 0으로 밀어낸다
        return -1 << (32 - prefix);
    }

    public static void main(String[] args) {
        String ip = "192.168.1.100";
        int prefix = 24;

        int ipInt = ipToInt(ip);
        int mask = cidrToMask(prefix);

        // 네트워크 주소 = IP & 마스크
        int network = ipInt & mask;

        // 브로드캐스트 주소 = 네트워크 | ~마스크
        int broadcast = network | ~mask;

        // 사용 가능한 호스트 수 = 브로드캐스트 - 네트워크 - 1
        int hostCount = (broadcast - network) - 1;

        System.out.println("IP:         " + ip);
        System.out.println("Mask:       " + intToIp(mask));       // 255.255.255.0
        System.out.println("Network:    " + intToIp(network));    // 192.168.1.0
        System.out.println("Broadcast:  " + intToIp(broadcast));  // 192.168.1.255
        System.out.println("Host count: " + hostCount);           // 254
    }
}
```

### Python

```python
import ipaddress

def subnet_calc(ip_str, prefix):
    # 수동 계산
    parts = [int(p) for p in ip_str.split('.')]
    ip = (parts[0] << 24) | (parts[1] << 16) | (parts[2] << 8) | parts[3]

    # 마스크: 상위 prefix개 비트가 1, 나머지 0
    mask = (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF

    network = ip & mask
    broadcast = network | (~mask & 0xFFFFFFFF)

    def to_ip(n):
        return f"{(n >> 24) & 0xFF}.{(n >> 16) & 0xFF}.{(n >> 8) & 0xFF}.{n & 0xFF}"

    return {
        "network": to_ip(network),
        "broadcast": to_ip(broadcast),
        "mask": to_ip(mask),
        "hosts": broadcast - network - 1,
    }

result = subnet_calc("10.0.5.130", 22)
# {'network': '10.0.4.0', 'broadcast': '10.0.7.255', 'mask': '255.255.252.0', 'hosts': 1022}
```

핵심 연산은 딱 두 줄이다:

- **네트워크 주소** = `IP & mask` — AND로 호스트 부분을 날린다
- **브로드캐스트 주소** = `network | ~mask` — OR로 호스트 부분을 전부 1로 채운다

Python에서는 `~mask`가 음수가 되기 때문에 `& 0xFFFFFFFF`로 32비트 범위를 맞춰야 한다. 실무에서는 `ipaddress` 모듈을 쓰는 게 실수가 적다.

## 실무 예제 3: XOR swap

임시 변수 없이 두 값을 교환하는 기법이다. XOR의 가역성(`a ^ b ^ b = a`)을 이용한다.

### Java

```java
int a = 5, b = 9;

a = a ^ b;  // a = 5^9
b = a ^ b;  // b = 5^9^9 = 5
a = a ^ b;  // a = 5^9^5 = 9

// a = 9, b = 5
```

### JavaScript

```javascript
let a = 0xFF, b = 0xAB;

a ^= b;
b ^= a;
a ^= b;

// a = 0xAB, b = 0xFF
```

### Python

```python
a, b = 42, 17

a ^= b
b ^= a
a ^= b

# a = 17, b = 42
```

**실무에서는 거의 안 쓴다.** 이유가 있다:

- `a`와 `b`가 같은 메모리 주소를 가리키면 둘 다 0이 된다
- 컴파일러 최적화가 더 나은 결과를 만든다
- 코드를 읽는 사람이 혼란스러워한다

면접에서 나오거나, 임베디드에서 레지스터가 극히 제한된 상황 정도에서만 의미가 있다.

## 실무 예제 4: 피처 플래그

서비스에서 기능 ON/OFF를 비트 플래그로 관리하는 패턴이다. DB에 정수 하나만 저장하면 된다.

### Java

```java
public class FeatureFlags {
    static final int DARK_MODE       = 1 << 0;
    static final int BETA_UI         = 1 << 1;
    static final int NEW_SEARCH      = 1 << 2;
    static final int EXPORT_CSV      = 1 << 3;
    static final int BULK_DELETE     = 1 << 4;

    private int flags;

    public FeatureFlags(int flags) {
        this.flags = flags;
    }

    public boolean isEnabled(int feature) {
        return (flags & feature) != 0;
    }

    public void enable(int feature) {
        flags |= feature;
    }

    public void disable(int feature) {
        flags &= ~feature;
    }

    // DB에 저장할 값
    public int toInt() {
        return flags;
    }

    // DB에서 불러오기
    public static FeatureFlags fromInt(int value) {
        return new FeatureFlags(value);
    }

    public static void main(String[] args) {
        // DB에서 flags = 5 (0b00101)로 저장되어 있었다면
        FeatureFlags user = FeatureFlags.fromInt(5);

        user.isEnabled(DARK_MODE);   // true  (bit 0)
        user.isEnabled(BETA_UI);     // false (bit 1)
        user.isEnabled(NEW_SEARCH);  // true  (bit 2)
    }
}
```

## 실무 예제 5: 비트 연산 유틸리티

자주 쓰이는 비트 트릭들을 모았다.

### 2의 거듭제곱 판별

```java
// n이 2의 거듭제곱이면 true
// 2의 거듭제곱은 비트가 딱 하나만 1이다 (예: 8 = 0b1000)
// n - 1 하면 그 아래 비트가 전부 1이 된다 (예: 7 = 0b0111)
// 둘을 AND 하면 0이 된다
boolean isPowerOfTwo(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}
```

```python
def is_power_of_two(n):
    return n > 0 and (n & (n - 1)) == 0

is_power_of_two(64)   # True
is_power_of_two(100)  # False
```

### 최하위 설정 비트(Lowest Set Bit) 추출

```java
// 가장 낮은 위치의 1 비트만 남긴다
// 12 = 0b1100 -> 4 = 0b0100
int lowestSetBit = n & (-n);
```

### 특정 비트 읽기/쓰기/끄기

```javascript
// n번째 비트 읽기
function getBit(value, n) {
  return (value >> n) & 1;
}

// n번째 비트 켜기
function setBit(value, n) {
  return value | (1 << n);
}

// n번째 비트 끄기
function clearBit(value, n) {
  return value & ~(1 << n);
}

// n번째 비트 토글
function toggleBit(value, n) {
  return value ^ (1 << n);
}

let v = 0b1010;
getBit(v, 1);      // 1
getBit(v, 0);      // 0
setBit(v, 0);      // 0b1011
clearBit(v, 3);    // 0b0010
toggleBit(v, 1);   // 0b1000
```

### RGB 색상 값 분리

32비트 정수에 ARGB 값이 들어있을 때 각 채널을 분리하는 코드다.

```java
int color = 0xFF5733FF; // ARGB

int alpha = (color >> 24) & 0xFF; // 0xFF = 255
int red   = (color >> 16) & 0xFF; // 0x57 = 87
int green = (color >> 8)  & 0xFF; // 0x33 = 51
int blue  = color & 0xFF;         // 0xFF = 255

// 채널 합치기
int merged = (alpha << 24) | (red << 16) | (green << 8) | blue;
```

```python
color = 0xFF5733FF

alpha = (color >> 24) & 0xFF
red   = (color >> 16) & 0xFF
green = (color >> 8) & 0xFF
blue  = color & 0xFF

# 합치기
merged = (alpha << 24) | (red << 16) | (green << 8) | blue
```

## 주의사항

### JavaScript의 비트 연산 함정

JavaScript의 비트 연산자는 피연산자를 **32비트 부호 있는 정수로 변환**한다. 큰 숫자를 다루면 의도와 다른 결과가 나온다.

```javascript
// 32비트 범위를 넘으면 잘린다
1 << 31  // -2147483648 (부호 비트가 켜짐)
1 << 32  // 1 (32비트 넘어가면 0으로 돌아감)

// BigInt를 써야 하는 경우
const big = 1n << 40n; // 1099511627776n
```

### Python의 정수 크기

Python은 정수 크기 제한이 없어서 `~` 연산 결과가 다른 언어와 다르다.

```python
~0   # -1 (무한 비트 기준으로 전부 1)
~5   # -6

# 32비트 기준으로 NOT을 하고 싶다면
(~5) & 0xFFFFFFFF  # 4294967290 = 0xFFFFFFFA
```

### 부호 있는 시프트

오른쪽 시프트에서 부호 비트 처리가 언어마다 다르다.

```java
// Java: >> 는 부호 유지, >>> 는 부호 무시
int n = -8;
n >> 1;   // -4 (부호 유지)
n >>> 1;  // 2147483644 (최상위에 0 채움)
```

```python
# Python: >> 는 항상 부호 유지 (>>> 없음)
-8 >> 1  # -4
```
