# 2진수를 16진수로 변환하는 방법

## 배경

진수는 수를 표현하는 방식으로, 우리가 일상에서 사용하는 10진수 외에도 2진수, 8진수, 16진수 등이 있습니다. 각 진수는 특정한 기수(radix)를 기반으로 합니다.

### 진수별 특징
- **2진수**: 기수 2 (0, 1) - 컴퓨터의 기본 언어
- **8진수**: 기수 8 (0~7) - Unix 권한 등에서 사용
- **10진수**: 기수 10 (0~9) - 일상생활에서 사용
- **16진수**: 기수 16 (0~9, A~F) - 컴퓨터 프로그래밍에서 널리 사용

### 16진수의 특징
- 16진수는 0부터 9까지의 숫자와 A부터 F까지의 알파벳으로 표현됩니다
- 각 자릿수는 4개의 비트를 나타내므로, 2진수와의 변환이 매우 용이합니다
- 컴퓨터 프로그래밍에서 메모리 주소나 색상 코드 등을 표현할 때 자주 사용됩니다

### 2진수-16진수 변환의 필요성
- **메모리 주소 표현**: 컴퓨터 메모리 주소는 보통 16진수로 표시
- **색상 코드**: 웹 개발에서 RGB 색상을 16진수로 표현
- **디버깅**: 기계어나 어셈블리 코드 분석 시 16진수 사용
- **데이터 압축**: 2진수를 더 간결하게 표현

## 핵심

### 2진수를 16진수로 변환하는 기본 원리

#### 1. 변환 과정
1. 2진수를 4자리씩 묶습니다
2. 각 묶음을 16진수로 변환합니다
3. 변환된 16진수를 순서대로 나열합니다

#### 2. 4자리 2진수 변환표
| 2진수 | 16진수 | 2진수 | 16진수 |
|-------|--------|-------|--------|
| 0000  | 0      | 1000  | 8      |
| 0001  | 1      | 1001  | 9      |
| 0010  | 2      | 1010  | A      |
| 0011  | 3      | 1011  | B      |
| 0100  | 4      | 1100  | C      |
| 0101  | 5      | 1101  | D      |
| 0110  | 6      | 1110  | E      |
| 0111  | 7      | 1111  | F      |

#### 3. 자릿수 맞추기
2진수의 자릿수가 4의 배수가 아닌 경우, 앞에 0을 채워서 4의 배수로 만듭니다.

```javascript
// 2진수 자릿수 맞추기 함수
function padBinary(binary) {
    const remainder = binary.length % 4;
    if (remainder !== 0) {
        const padding = '0'.repeat(4 - remainder);
        return padding + binary;
    }
    return binary;
}

// 사용 예시
console.log(padBinary('1011'));     // "00001011"
console.log(padBinary('110101'));   // "00110101"
console.log(padBinary('11110000')); // "11110000" (이미 4의 배수)
```

### 수동 변환 방법

#### 단계별 변환 과정
```javascript
// 수동 변환 함수
function manualBinaryToHex(binary) {
    // 1. 자릿수 맞추기
    const padded = padBinary(binary);
    
    // 2. 4자리씩 묶기
    const groups = [];
    for (let i = 0; i < padded.length; i += 4) {
        groups.push(padded.slice(i, i + 4));
    }
    
    // 3. 각 묶음을 16진수로 변환
    const hexGroups = groups.map(group => {
        const decimal = parseInt(group, 2);
        return decimal.toString(16).toUpperCase();
    });
    
    // 4. 결과 조합
    return '0x' + hexGroups.join('');
}

// 변환 과정 시각화
function visualizeConversion(binary) {
    console.log('변환 과정:');
    console.log('1. 원본 2진수:', binary);
    
    const padded = padBinary(binary);
    console.log('2. 자릿수 맞춤:', padded);
    
    const groups = [];
    for (let i = 0; i < padded.length; i += 4) {
        groups.push(padded.slice(i, i + 4));
    }
    console.log('3. 4자리씩 묶기:', groups.join(' '));

        const decimal = parseInt(group, 2);
        const hex = decimal.toString(16).toUpperCase();
        console.log(`   ${group} → ${decimal} → ${hex}`);
        return hex;
    });
    
    const result = '0x' + hexGroups.join('');
    console.log('4. 최종 결과:', result);
    
    return result;
}
```

```javascript
function binaryToHex(binary) {
    // 입력 검증
    if (!/^[01]+$/.test(binary)) {
        throw new Error('유효하지 않은 2진수입니다.');
    }
    
    // 자릿수 맞추기
    const remainder = binary.length % 4;
    const padded = remainder === 0 ? binary : '0'.repeat(4 - remainder) + binary;
    
    // 4자리씩 묶어서 변환
    let hex = '';
    for (let i = 0; i < padded.length; i += 4) {
        const group = padded.slice(i, i + 4);
        const hexDigit = parseInt(group, 2).toString(16).toUpperCase();
        hex += hexDigit;
    }
    
    return `0x${hex}`;
}

// 사용 예시
console.log(binaryToHex('10110101'));      // "0xB5"
console.log(binaryToHex('110011001100'));  // "0xCCC"
console.log(binaryToHex('1111000011110000')); // "0xF0F0"
```

#### 고급 변환 함수 (에러 처리 포함)
```javascript
class BinaryConverter {
    static toHex(binary) {
        try {
            // 입력 검증
            if (typeof binary !== 'string') {
                throw new Error('2진수는 문자열로 입력해야 합니다.');
            }
            
            if (!/^[01]+$/.test(binary)) {
                throw new Error('유효하지 않은 2진수입니다. 0과 1만 사용하세요.');
            }
            
            if (binary.length === 0) {
                throw new Error('빈 문자열은 변환할 수 없습니다.');
            }
            
            // 자릿수 맞추기
            const remainder = binary.length % 4;
            const padded = remainder === 0 ? binary : '0'.repeat(4 - remainder) + binary;
            
            // 4자리씩 묶어서 변환
            let hex = '';
            for (let i = 0; i < padded.length; i += 4) {
                const group = padded.slice(i, i + 4);
                const hexDigit = parseInt(group, 2).toString(16).toUpperCase();
                hex += hexDigit;
            }
            
            return `0x${hex}`;
            
        } catch (error) {
            console.error('변환 오류:', error.message);
            return null;
        }
    }
    
    static toDecimal(binary) {
        return parseInt(binary, 2);
    }
    
    static validate(binary) {
        return /^[01]+$/.test(binary);
    }
    
    // 변환 과정 상세 출력
    static toHexWithSteps(binary) {
        if (!this.validate(binary)) {
            return null;
        }
        
        const steps = {
            original: binary,
            padded: padBinary(binary),
            groups: [],
            conversions: [],
            result: ''
        };
        
        const padded = steps.padded;
        for (let i = 0; i < padded.length; i += 4) {
            const group = padded.slice(i, i + 4);
            steps.groups.push(group);
            
            const decimal = parseInt(group, 2);
            const hex = decimal.toString(16).toUpperCase();
            steps.conversions.push(`${group} → ${decimal} → ${hex}`);
        }
        
        steps.result = '0x' + steps.groups.map(group => 
            parseInt(group, 2).toString(16).toUpperCase()
        ).join('');
        
        return steps;
    }
}

// 사용 예시
console.log(BinaryConverter.toHex('10110101'));      // "0xB5"
console.log(BinaryConverter.toHex('110011001100'));  // "0xCCC"
console.log(BinaryConverter.toHex('invalid'));       // null (오류)
console.log(BinaryConverter.validate('10110101'));   // true
console.log(BinaryConverter.validate('invalid'));    // false

// 상세 변환 과정
const steps = BinaryConverter.toHexWithSteps('10110101');
console.log('변환 과정:', steps);
```

### Python을 사용한 변환 함수

#### Python 변환 함수
```python
def binary_to_hex(binary):
    """
    2진수를 16진수로 변환하는 함수
    
    Args:
        binary (str): 2진수 문자열
        
    Returns:
        str: 16진수 문자열 (0x 접두사 포함)
    """
    try:
        # 입력 검증
        if not isinstance(binary, str):
            raise ValueError("2진수는 문자열로 입력해야 합니다.")
        
        if not all(bit in '01' for bit in binary):
            raise ValueError("유효하지 않은 2진수입니다. 0과 1만 사용하세요.")
        
        if len(binary) == 0:
            raise ValueError("빈 문자열은 변환할 수 없습니다.")
        
        # 자릿수 맞추기
        remainder = len(binary) % 4
        if remainder != 0:
            binary = '0' * (4 - remainder) + binary
        
        # 4자리씩 묶어서 변환
        hex_result = ''
        for i in range(0, len(binary), 4):
            group = binary[i:i+4]
            hex_digit = hex(int(group, 2))[2:].upper()
            hex_result += hex_digit
        
        return f"0x{hex_result}"
        
    except Exception as e:
        print(f"변환 오류: {e}")
        return None

# 사용 예시
print(binary_to_hex('10110101'))      # "0xB5"
print(binary_to_hex('110011001100'))  # "0xCCC"
print(binary_to_hex('1111000011110000')) # "0xF0F0"
print(binary_to_hex('invalid'))       # None (오류)
```

```javascript
class OptimizedBinaryConverter {
    constructor() {
        // 4자리 2진수 변환 캐시
        this.cache = new Map();
        this.initializeCache();
    }
    
    initializeCache() {
        for (let i = 0; i < 16; i++) {
            const binary = i.toString(2).padStart(4, '0');
            const hex = i.toString(16).toUpperCase();
            this.cache.set(binary, hex);
        }
    }
    
    toHex(binary) {
        // 자릿수 맞추기
        const remainder = binary.length % 4;
        const padded = remainder === 0 ? binary : '0'.repeat(4 - remainder) + binary;
        
        // 캐시를 사용하여 변환
        let hex = '';
        for (let i = 0; i < padded.length; i += 4) {
            const group = padded.slice(i, i + 4);
            hex += this.cache.get(group);
        }
        
        return `0x${hex}`;
    }
    
    // 성능 측정
    benchmark(binary, iterations = 100000) {
        console.time('캐시 사용');
        for (let i = 0; i < iterations; i++) {
            this.toHex(binary);
        }
        console.timeEnd('캐시 사용');
        
        console.time('일반 변환');
        for (let i = 0; i < iterations; i++) {
            BinaryConverter.toHex(binary);
        }
        console.timeEnd('일반 변환');
    }
}

// 사용 예시

// 성능 비교
```

### 메모리 효율성

#### 스트림 기반 변환 (대용량 데이터)
```javascript
class StreamBinaryConverter {
    constructor() {
        this.buffer = '';
        this.cache = new Map();
        this.initializeCache();
    }
    
    initializeCache() {
        for (let i = 0; i < 16; i++) {
            const binary = i.toString(2).padStart(4, '0');
            const hex = i.toString(16).toUpperCase();
            this.cache.set(binary, hex);
        }
    }
    
    // 스트림으로 2진수 입력
    write(binary) {
        this.buffer += binary;
    }
    
    // 완성된 4자리 그룹을 16진수로 변환
    *read() {
        while (this.buffer.length >= 4) {
            const group = this.buffer.slice(0, 4);
            this.buffer = this.buffer.slice(4);
            yield this.cache.get(group);
        }
    }
    
    // 남은 버퍼 처리
    flush() {
        if (this.buffer.length > 0) {
            const padded = this.buffer.padEnd(4, '0');
            const hex = this.cache.get(padded);
            this.buffer = '';
            return hex;
        }
        return '';
    }
}

// 사용 예시
const streamConverter = new StreamBinaryConverter();
streamConverter.write('10110101');
streamConverter.write('11001100');

const hexDigits = [];
for (const digit of streamConverter.read()) {
    hexDigits.push(digit);
}

const remaining = streamConverter.flush();
if (remaining) {
    hexDigits.push(remaining);
}

console.log('스트림 변환 결과:', '0x' + hexDigits.join(''));
```

```javascript
function validateBinary(binary) {
    const errors = [];
    
    // 타입 검사
    if (typeof binary !== 'string') {
        errors.push('2진수는 문자열로 입력해야 합니다.');
        return { isValid: false, errors };
    }
    
    // 길이 검사
    if (binary.length === 0) {
        errors.push('빈 문자열은 변환할 수 없습니다.');
        return { isValid: false, errors };
    }
    
    // 문자 검사
    if (!/^[01]+$/.test(binary)) {
        errors.push('유효하지 않은 2진수입니다. 0과 1만 사용하세요.');
        return { isValid: false, errors };
    }
    
    // 길이 제한 (메모리 보호)
    if (binary.length > 1000000) {
        errors.push('2진수가 너무 깁니다. 최대 1,000,000자까지 지원합니다.');
        return { isValid: false, errors };
    }
    
    return { isValid: true, errors: [] };
}

// 테스트 케이스
const testCases = [
    '10110101',
    'invalid',
    '',
    123,
    '1'.repeat(1000001)
];

testCases.forEach(test => {
    const validation = validateBinary(test);
    if (validation.isValid) {
        console.log(`${test}는 유효한 2진수입니다.`);
        console.log('변환 결과:', BinaryConverter.toHex(test));
    } else {
        console.log(`${test}는 유효하지 않습니다:`, validation.errors);
    }
});
```

```javascript
function hexToRgb(hex) {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16)
    } : null;
}

// RGB를 16진수로 변환
function rgbToHex(r, g, b) {
    return '#' + [r, g, b].map(x => {
        const hex = x.toString(16);
        return hex.length === 1 ? '0' + hex : hex;
    }).join('');
}
```

```javascript
const mask = 0xFF;  // 11111111 (8비트 모두 1)
const value = 0xAB; // 10101011

console.log((value & mask).toString(16));  // "ab" (마스킹 결과)
console.log((value | mask).toString(16));  // "ff" (OR 연산)

// 플래그 설정
const FLAG_READ = 0x01;    // 0001
const FLAG_WRITE = 0x02;   // 0010
const FLAG_EXECUTE = 0x04; // 0100

let permissions = FLAG_READ | FLAG_WRITE;  // 0011
console.log('권한:', permissions.toString(16));  // "3"
```

### 다른 진수와의 변환

#### 10진수를 16진수로 변환
```javascript
function decimalToHex(decimal) {
    return `0x${decimal.toString(16).toUpperCase()}`;
}

console.log(decimalToHex(181));  // "0xB5"
console.log(decimalToHex(255));  // "0xFF"
```

#### 16진수를 2진수로 변환
```javascript
function hexToBinary(hex) {
    // 0x 접두사 제거
    const cleanHex = hex.replace(/^0x/i, '');
    const decimal = parseInt(cleanHex, 16);
    return decimal.toString(2);
}

console.log(hexToBinary('0xB5'));  // "10110101"
console.log(hexToBinary('0xFF'));  // "11111111"
```

2진수를 16진수로 변환하는 방법은 컴퓨터 과학의 기본 개념 중 하나입니다.
