# JavaScript 논리 연산자 개념과 사용법

## 배경

JavaScript에서 논리 연산자는 조건문을 결합하거나 값을 평가하는 데 사용되는 중요한 연산자들입니다. 주요 논리 연산자로는 &&(AND), ||(OR), !(NOT)이 있으며, 단축 평가(Short-circuit evaluation)를 통해 효율적인 코드를 작성할 수 있습니다.

### 논리 연산자의 필요성
- **조건 결합**: 여러 조건을 논리적으로 결합
- **값 평가**: 표현식의 참/거짓 판단
- **단축 평가**: 불필요한 연산 방지
- **기본값 설정**: falsy 값 처리

### 기본 개념
- **AND (&&)**: 모든 조건이 참일 때만 참
- **OR (||)**: 하나라도 참이면 참
- **NOT (!)**: 값을 반전
- **단축 평가**: 첫 번째 피연산자로 결과 결정

## 핵심

### 1. AND 연산자 (&&)

#### 기본 동작
```javascript
// 두 개 이상의 조건이 모두 참일 때만 true 반환
const a = true;
const b = false;
const c = true;

console.log(a && b);    // false
console.log(a && c);    // true
console.log(b && c);    // false
console.log(a && b && c); // false
```

#### 단축 평가
```javascript
// 왼쪽 피연산자가 false이면 오른쪽은 평가하지 않음
const result = false && someFunction(); // someFunction()은 호출되지 않음

// 실제 사용 예시
const user = null;
const name = user && user.name; // user가 null이므로 user.name은 평가되지 않음
console.log(name); // null
```

#### 실제 활용
```javascript
// 조건부 렌더링
const isAdmin = true;
const isLoggedIn = true;

if (isAdmin && isLoggedIn) {
    console.log("관리자 대시보드에 접근합니다.");
}

// 안전한 객체 속성 접근
const user = {
    profile: {
        name: 'Alice',
        age: 25
    }
};

const userName = user && user.profile && user.profile.name;
console.log(userName); // 'Alice'

// 조건부 함수 실행
function showAdminPanel() {
    console.log('관리자 패널 표시');
}

const isAdmin = true;
const hasPermission = true;

isAdmin && hasPermission && showAdminPanel(); // 조건이 모두 참일 때만 실행
```

### 2. OR 연산자 (||)

#### 기본 동작
```javascript
// 여러 조건 중 하나라도 참이면 true 반환
const isRaining = false;
const isSnowing = true;
const isCloudy = false;

console.log(isRaining || isSnowing);    // true
console.log(isRaining || isCloudy);     // false
console.log(isRaining || isSnowing || isCloudy); // true
```

#### 단축 평가
```javascript
// 왼쪽 피연산자가 true이면 오른쪽은 평가하지 않음
const result = true || someFunction(); // someFunction()은 호출되지 않음

// 실제 사용 예시
const defaultName = "Guest";
const userName = user.name || defaultName; // user.name이 falsy면 defaultName 사용
```

#### 실제 활용
```javascript
// 기본값 설정
const userName = user.name || "익명 사용자";
const userAge = user.age || 0;
const userEmail = user.email || "이메일 없음";

// 조건부 실행
function showWelcomeMessage() {
    console.log('환영합니다!');
}

function showLoginMessage() {
    console.log('로그인이 필요합니다.');
}

const isLoggedIn = false;
isLoggedIn ? showWelcomeMessage() : showLoginMessage();

// 또는 OR 연산자 사용
(isLoggedIn && showWelcomeMessage) || showLoginMessage();
```

### 3. NOT 연산자 (!)

#### 기본 동작
```javascript
// 값을 반전시킴 (true는 false로, false는 true로)
const isLoggedIn = true;
console.log(!isLoggedIn);     // false
console.log(!!isLoggedIn);    // true (이중 부정)

// 다양한 값에 대한 NOT 연산
console.log(!true);      // false
console.log(!false);     // true
console.log(!0);         // true
console.log(!1);         // false
console.log(!"");        // true
console.log(!"hello");   // false
console.log(!null);      // true
console.log(!undefined); // true
console.log(!{});        // false
console.log(![]);        // false
```

#### 실제 활용
```javascript
// 조건 반전
const isDisabled = false;
const isEnabled = !isDisabled; // true

// 필수 입력 검증
const username = "";
const password = "123456";

if (!username || !password) {
    console.log("사용자명과 비밀번호를 모두 입력해주세요.");
}

// 존재 여부 확인
const user = null;
if (!user) {
    console.log("사용자를 찾을 수 없습니다.");
}

// 타입 변환 (boolean으로)
const truthyValue = "hello";
const falsyValue = "";

console.log(!!truthyValue); // true
console.log(!!falsyValue);  // false
```

### 4. 복합 논리 연산

#### 우선순위
```javascript
// NOT > AND > OR 순서로 평가
const result = !true && false || true;
// 1. !true = false
// 2. false && false = false
// 3. false || true = true
console.log(result); // true

// 괄호를 사용하여 명확하게 표현
const result2 = (!true && false) || true;
console.log(result2); // true
```

#### 복잡한 조건
```javascript
const age = 20;
const hasID = true;
const hasPermission = true;
const isVIP = false;

// 복합 조건
if (age >= 18 && hasID && (hasPermission || isVIP)) {
    console.log("접근이 허용됩니다.");
}

// 가독성을 위한 변수 분리
const isAdult = age >= 18;
const hasValidID = hasID;
const hasAccess = hasPermission || isVIP;

if (isAdult && hasValidID && hasAccess) {
    console.log("접근이 허용됩니다.");
}
```

## 예시

### 1. 실제 사용 사례

#### 폼 검증
```javascript
function validateForm(formData) {
    const { username, email, password, confirmPassword } = formData;
    
    // 각 필드 검증
    const isUsernameValid = username && username.length >= 3;
    const isEmailValid = email && email.includes('@');
    const isPasswordValid = password && password.length >= 6;
    const doPasswordsMatch = password === confirmPassword;
    
    // 전체 폼 유효성 검사
    const isFormValid = isUsernameValid && isEmailValid && isPasswordValid && doPasswordsMatch;
    
    if (!isFormValid) {
        console.log('폼 검증 실패:');
        if (!isUsernameValid) console.log('- 사용자명은 3자 이상이어야 합니다.');
        if (!isEmailValid) console.log('- 유효한 이메일을 입력해주세요.');
        if (!isPasswordValid) console.log('- 비밀번호는 6자 이상이어야 합니다.');
        if (!doPasswordsMatch) console.log('- 비밀번호가 일치하지 않습니다.');
    }
    
    return isFormValid;
}

// 사용 예시
const formData = {
    username: 'alice',
    email: 'alice@example.com',
    password: '123456',
    confirmPassword: '123456'
};

const isValid = validateForm(formData);
console.log('폼 유효성:', isValid);
```

#### 권한 검사
```javascript
function checkPermission(user, action) {
    // 사용자 기본 권한
    const hasBasicPermission = user && user.isActive;
    
    // 관리자 권한
    const isAdmin = user && user.role === 'admin';
    
    // 특정 권한
    const hasSpecificPermission = user && user.permissions && user.permissions.includes(action);
    
    // VIP 사용자
    const isVIP = user && user.isVIP;
    
    // 최종 권한 결정
    const hasPermission = hasBasicPermission && (isAdmin || hasSpecificPermission || isVIP);
    
    return hasPermission;
}

// 사용 예시
const user = {
    id: 1,
    name: 'Alice',
    isActive: true,
    role: 'user',
    permissions: ['read', 'write'],
    isVIP: false
};

console.log('읽기 권한:', checkPermission(user, 'read')); // true
console.log('삭제 권한:', checkPermission(user, 'delete')); // false
```

### 2. 고급 패턴

#### 조건부 객체 속성
```javascript
function createUserConfig(user) {
    return {
        name: user.name || 'Anonymous',
        email: user.email || 'no-email@example.com',
        role: user.role || 'user',
        // 조건부 속성 추가
        ...(user.isAdmin && { permissions: ['all'] }),
        ...(user.isVIP && { vipLevel: user.vipLevel }),
        ...(user.avatar && { avatarUrl: user.avatar })
    };
}

// 사용 예시
const user1 = { name: 'Alice', isAdmin: true };
const user2 = { name: 'Bob', isVIP: true, vipLevel: 2, avatar: 'avatar.jpg' };

console.log(createUserConfig(user1));
console.log(createUserConfig(user2));
```

#### 함수형 프로그래밍
```javascript
// 조건부 함수 실행
function executeIf(condition, fn) {
    return condition && fn();
}

function logMessage(message) {
    console.log(message);
}

// 사용 예시
const isDebugMode = true;
const isProduction = false;

executeIf(isDebugMode, () => logMessage('디버그 모드입니다.'));
executeIf(isProduction, () => logMessage('프로덕션 모드입니다.'));

// 파이프라인에서의 사용
function processData(data) {
    return data
        .filter(item => item && item.isValid)
        .map(item => ({
            ...item,
            processed: true,
            timestamp: Date.now()
        }))
        .filter(item => item.timestamp > 0);
}
```

## 운영 팁

### 성능 최적화

#### 단축 평가 활용
```javascript
// 비효율적인 방법
function expensiveOperation() {
    console.log('비용이 많이 드는 연산');
    return true;
}

// 조건이 false인 경우에도 실행됨
if (false && expensiveOperation()) {
    console.log('실행되지 않음');
}

// 효율적인 방법: 단축 평가 활용
if (false && expensiveOperation()) {
    console.log('실행되지 않음');
}
// expensiveOperation()은 호출되지 않음
```

#### 조건 최적화
```javascript
// 비효율적인 조건 순서
function checkCondition() {
    console.log('조건 검사');
    return false;
}

function checkAnotherCondition() {
    console.log('다른 조건 검사');
    return true;
}

// false가 먼저 나오는 경우
if (checkCondition() && checkAnotherCondition()) {
    console.log('실행됨');
}

// true가 먼저 나오는 경우
if (checkAnotherCondition() && checkCondition()) {
    console.log('실행됨');
}
```

### 에러 처리

#### 안전한 속성 접근
```javascript
// 안전하지 않은 방법
function getUserName(user) {
    return user.name; // user가 null이면 오류 발생
}

// 안전한 방법
function getUserNameSafe(user) {
    return user && user.name;
}

// 더 안전한 방법
function getUserNameSafest(user) {
    return user && user.name || 'Unknown';
}

// 사용 예시
const user = null;
console.log(getUserNameSafe(user)); // null
console.log(getUserNameSafest(user)); // 'Unknown'
```

#### 조건부 함수 실행
```javascript
// 안전한 함수 호출
function safeExecute(fn, ...args) {
    return typeof fn === 'function' && fn(...args);
}

// 사용 예시
const callback = null;
const validCallback = () => console.log('실행됨');

safeExecute(callback); // 아무것도 실행되지 않음
safeExecute(validCallback); // '실행됨' 출력
```

## 참고

### 논리 연산자 우선순위

| 연산자 | 우선순위 | 설명 |
|--------|----------|------|
| **!** | 1 | NOT (가장 높음) |
| **&&** | 2 | AND |
| **\|\|** | 3 | OR (가장 낮음) |

### Falsy 값들

| 값 | 타입 | 설명 |
|----|------|------|
| **false** | Boolean | 거짓 |
| **0** | Number | 숫자 0 |
| **-0** | Number | 음수 0 |
| **0n** | BigInt | BigInt 0 |
| **""** | String | 빈 문자열 |
| **null** | null | null 값 |
| **undefined** | undefined | 정의되지 않음 |
| **NaN** | Number | 숫자가 아님 |

### 논리 연산자 사용 권장사항

| 상황 | 권장사항 | 이유 |
|------|----------|------|
| **조건 결합** | && 사용 | 모든 조건 만족 필요 |
| **기본값 설정** | \|\| 사용 | falsy 값 처리 |
| **조건 반전** | ! 사용 | 명확한 부정 표현 |
| **복잡한 조건** | 괄호 사용 | 우선순위 명확화 |
| **성능 중요** | 단축 평가 활용 | 불필요한 연산 방지 |
